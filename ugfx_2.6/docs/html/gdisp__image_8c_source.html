99 gdispImageError gdispImageOpenGFile(gdispImage *img, GFILE *f) {
100  gdispImageError err;
101 
102  if (!f)
103  return GDISP_IMAGE_ERR_NOSUCHFILE;
104  img->f = f;
105  img->bgcolor = White;
106  for(img->fns = ImageHandlers; img->fns < ImageHandlers+sizeof(ImageHandlers)/sizeof(ImageHandlers[0]); img->fns++) {
107  err = img->fns->open(img);
108  if (err != GDISP_IMAGE_ERR_BADFORMAT) {
109  if ((err & GDISP_IMAGE_ERR_UNRECOVERABLE))
110  goto unrecoverable;
111 
112  // Everything is possible
113  return err;
114  }
115 
116  // Try the next decoder
117  gfileSetPos(img->f, 0);
118  }
119 
120  err = GDISP_IMAGE_ERR_BADFORMAT;
121  img->type = GDISP_IMAGE_TYPE_UNKNOWN;
122 
123 unrecoverable:
124  gfileClose(img->f);
125  img->f = 0;
126  img->flags = 0;
127  img->fns = 0;
128  img->priv = 0;
129  return err;
130 }

155 gdispImageError gdispGImageDraw(GDisplay *g, gdispImage *img, coord_t x, coord_t y, coord_t cx, coord_t cy, coord_t sx, coord_t sy) {
156  if (!img->fns) return GDISP_IMAGE_ERR_BADFORMAT;
157 
158  // Check on window
159  if (cx <= 0 || cy <= 0) return GDISP_IMAGE_ERR_OK;
160  if (sx < 0) sx = 0;
161  if (sy < 0) sy = 0;
162  if (sx >= img->width || sy >= img->height) return GDISP_IMAGE_ERR_OK;
163  if (sx + cx > img->width) cx = img->width - sx;
164  if (sy + cy > img->height) cy = img->height - sy;
165 
166  // Draw
167  return img->fns->draw(g, img, x, y, cx, cy, sx, sy);
168 }
