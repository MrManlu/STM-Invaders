65  void _gwinUpdate(GHandle gh) {
66  if ((gh->flags & GWIN_FLG_SYSVISIBLE)) {
67  if (gh->vmt->Redraw) {
68  getLock(gh);
69  gh->vmt->Redraw(gh);
70  exitLock(gh);
71  } else if ((gh->flags & GWIN_FLG_BGREDRAW)) {
72  getLock(gh);
73  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
74  exitLock(gh);
75  if (gh->vmt->AfterClear)
76  gh->vmt->AfterClear(gh);
77  }
78  } else if ((gh->flags & GWIN_FLG_BGREDRAW)) {
79  getLock(gh);
80  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gwinGetDefaultBgColor());
81  exitLock(gh);
82  }
83  gh->flags &= ~(GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW);

225 void _gwinFlushRedraws(GRedrawMethod how) {
226  GHandle gh;
227 
228  // Do we really need to do anything?
229  if (!RedrawPending)
230  return;
231 
232  // Obtain the drawing lock
233  if (how == REDRAW_WAIT)
234  gfxSemWait(&gwinsem, TIME_INFINITE);
235  else if (how == REDRAW_NOWAIT && !gfxSemWait(&gwinsem, TIME_IMMEDIATE))
236  // Someone is drawing - They will do the redraw when they are finished
237  return;
238 
239  // Do loss of visibility first
240  while ((RedrawPending & DOREDRAW_INVISIBLES)) {
241  RedrawPending &= ~DOREDRAW_INVISIBLES; // Catch new requests
242 
243  for(gh = gwinGetNextWindow(0); gh; gh = gwinGetNextWindow(gh)) {
244  if ((gh->flags & (GWIN_FLG_NEEDREDRAW|GWIN_FLG_SYSVISIBLE)) != GWIN_FLG_NEEDREDRAW)
245  continue;
246 
247  // Do the redraw
248  #if GDISP_NEED_CLIP
249  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
250  _GWINwm->vmt->Redraw(gh);
251  gdispGUnsetClip(gh->display);
252  #else
253  _GWINwm->vmt->Redraw(gh);
254  #endif
255 
256  // Postpone further redraws
257  #if !GWIN_REDRAW_IMMEDIATE && !GWIN_REDRAW_SINGLEOP
258  if (how == REDRAW_NOWAIT) {
259  RedrawPending |= DOREDRAW_INVISIBLES;
260  TriggerRedraw();
261  goto releaselock;
262  }
263  #endif
264  }
265  }
266 
267  // Do the visible windows next
268  while ((RedrawPending & DOREDRAW_VISIBLES)) {
269  RedrawPending &= ~DOREDRAW_VISIBLES; // Catch new requests
270 
271  for(gh = gwinGetNextWindow(0); gh; gh = gwinGetNextWindow(gh)) {
272  if ((gh->flags & (GWIN_FLG_NEEDREDRAW|GWIN_FLG_SYSVISIBLE)) != (GWIN_FLG_NEEDREDRAW|GWIN_FLG_SYSVISIBLE))
273  continue;
274 
275  // Do the redraw
276  #if GDISP_NEED_CLIP
277  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
278  _GWINwm->vmt->Redraw(gh);
279  gdispGUnsetClip(gh->display);
280  #else
281  _GWINwm->vmt->Redraw(gh);
282  #endif
283 
284  // Postpone further redraws (if there are any and the options are set right)
285  #if !GWIN_REDRAW_IMMEDIATE && !GWIN_REDRAW_SINGLEOP
286  if (how == REDRAW_NOWAIT) {
287  while((gh = gwinGetNextWindow(gh))) {
288  if ((gh->flags & (GWIN_FLG_NEEDREDRAW|GWIN_FLG_SYSVISIBLE)) == (GWIN_FLG_NEEDREDRAW|GWIN_FLG_SYSVISIBLE)) {
289  RedrawPending |= DOREDRAW_VISIBLES;
290  TriggerRedraw();
291  break;
292  }
293  }
294  goto releaselock;
295  }
296  #endif
297  }
298  }
299 
300  #if !GWIN_REDRAW_IMMEDIATE && !GWIN_REDRAW_SINGLEOP
301  releaselock:
302  #endif
303 
304  // Release the lock
305  if (how == REDRAW_WAIT || how == REDRAW_NOWAIT)
306  gfxSemSignal(&gwinsem);
307 }

322 #if GWIN_NEED_CONTAINERS
323  void _gwinRippleVisibility(void) {
324  GHandle gh;
325 
326  // Check each window's visibility is consistent with its parents
327  for(gh = gwinGetNextWindow(0); gh; gh = gwinGetNextWindow(gh)) {
328  switch(gh->flags & (GWIN_FLG_SYSVISIBLE|GWIN_FLG_VISIBLE)) {
329  case GWIN_FLG_VISIBLE:
330  if (!gh->parent || (gh->parent->flags & GWIN_FLG_SYSVISIBLE)) {
331  // We have been made visible
332  gh->flags |= (GWIN_FLG_SYSVISIBLE|GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW);
333 
334  // Do we want to grab the focus
335  _gwinFixFocus(gh);
336 
337  RedrawPending |= DOREDRAW_VISIBLES;
338  }
339  break;
340  case (GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE):
341  if (!gh->parent || (gh->parent->flags & GWIN_FLG_SYSVISIBLE))
342  break;
343 
344  // Parent has been made invisible
345  gh->flags &= ~GWIN_FLG_SYSVISIBLE;
346 
347  // No focus for us anymore
348  _gwinFixFocus(gh);
349 
350  break;
351  case GWIN_FLG_SYSVISIBLE:
352  // We have been made invisible
353  gh->flags &= ~GWIN_FLG_SYSVISIBLE;
354  if (!gh->parent || (gh->parent->flags & GWIN_FLG_SYSVISIBLE)) {
355  // The parent is visible so we must clear the area we took
356  gh->flags |= (GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW);
357 
358  // No focus for us anymore
359  _gwinFixFocus(gh);
360 
361  RedrawPending |= DOREDRAW_INVISIBLES;
362  }
363  break;
364  }
365  }
366  }
367 #endif

369 bool_t _gwinDrawStart(GHandle gh) {
370  // This test should occur inside the lock. We do this
371  // here as well as an early out (more efficient).
372  if (!(gh->flags & GWIN_FLG_SYSVISIBLE))
373  return FALSE;
374 
375  // Obtain the drawing lock
376  gfxSemWait(&gwinsem, TIME_INFINITE);
377 
378  // Re-test visibility as we may have waited a while
379  if (!(gh->flags & GWIN_FLG_SYSVISIBLE)) {
380  _gwinDrawEnd(gh);
381  return FALSE;
382  }
383 
384  // OK - we are ready to draw.
385  #if GDISP_NEED_CLIP
386  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
387  #endif
388  return TRUE;
389 }

634  void gwinSetFlashing(GHandle gh, bool_t flash) {
635 
636  // Start flashing?
637  if (flash) {
638  gh->flags |= GWIN_FLG_FLASHING; // A redraw will occur on the next flash period.
639 
640  // Start the flash timer if needed
641  if (!(RedrawPending & DOREDRAW_FLASHRUNNING)) {
642  RedrawPending |= DOREDRAW_FLASHRUNNING;
643 
644  // Ensure we start the timer with flash bit on
645  _gwinFlashState = FALSE;
646  FlashTimerFn(0); // First flash
647  gtimerStart(&FlashTimer, FlashTimerFn, 0, TRUE, GWIN_FLASHING_PERIOD); // Subsequent flashes
648  }
649

748 static void WM_Redraw(GHandle gh) {
749  #if GWIN_NEED_CONTAINERS
750  redo_redraw:
751  #endif
752  if ((gh->flags & GWIN_FLG_SYSVISIBLE)) {
753  if (gh->vmt->Redraw)
754  gh->vmt->Redraw(gh);
755  else if ((gh->flags & GWIN_FLG_BGREDRAW)) {
756  // We can't redraw but we want full coverage so just clear the area
757  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
758 
759  // Only do an after clear if this is not a parent reveal
760  if (!(gh->flags & GWIN_FLG_PARENTREVEAL) && gh->vmt->AfterClear)
761  gh->vmt->AfterClear(gh);
762  }
763 
764  #if GWIN_NEED_CONTAINERS
765  // If this is container but not a parent reveal, mark any visible children for redraw
766  // We redraw our children here as we have overwritten them in redrawing the parent
767  // as GDISP/GWIN doesn't support complex clipping regions.
768  if ((gh->flags & (GWIN_FLG_CONTAINER|GWIN_FLG_PARENTREVEAL)) == GWIN_FLG_CONTAINER) {
769 
770  // Container redraw is done
771  gh->flags &= ~(GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW|GWIN_FLG_PARENTREVEAL);
772 
773  for(gh = gwinGetFirstChild(gh); gh; gh = gwinGetSibling(gh))
774  _gwinUpdate(gh);
775  return;
776  }
777  #endif
778 
779  } else {
780  if ((gh->flags & GWIN_FLG_BGREDRAW)) {
781  GHandle gx;
782 
783  #if GWIN_NEED_CONTAINERS
784  if (gh->parent) {
785  // Child redraw is done
786  gh->flags &= ~(GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW|GWIN_FLG_PARENTREVEAL);
787 
788 
789  // Get the parent to redraw the area
790  gh = gh->parent;
791 
792  // The parent is already marked for redraw - don't do it now.
793  if ((gh->flags & GWIN_FLG_NEEDREDRAW))
794  return;
795 
796  // Use the existing clipping region and redraw now
797  gh->flags |= (GWIN_FLG_BGREDRAW|GWIN_FLG_PARENTREVEAL);
798  goto redo_redraw;
799  }
800  #endif
801 
802  // Clear the area to the background color
803  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gwinGetDefaultBgColor());
804 
805  // Now loop over all windows looking for overlaps. Redraw them if they overlap the newly exposed area.
806  for(gx = gwinGetNextWindow(0); gx; gx = gwinGetNextWindow(gx)) {
807  if ((gx->flags & GWIN_FLG_SYSVISIBLE)
808  && gx->display == gh->display
809  && gx->x < gh->x+gh->width && gx->y < gh->y+gh->height && gx->x+gx->width >= gh->x && gx->y+gx->height >= gh->y) {
810  if (gx->vmt->Redraw)
811  gx->vmt->Redraw(gx);
812  else
813  // We can't redraw this window but we want full coverage so just clear the area
814  gdispGFillArea(gx->display, gx->x, gx->y, gx->width, gx->height, gx->bgcolor);
815  }
816  }
817 
818  }
819  }
820 
821  // Redraw is done
822  gh->flags &= ~(GWIN_FLG_NEEDREDRAW|GWIN_FLG_BGREDRAW|GWIN_FLG_PARENTREVEAL);
823 }

825 static void WM_Size(GHandle gh, coord_t w, coord_t h) {
826  coord_t v;
827 
828  #if GWIN_NEED_CONTAINERS
829  if (gh->parent) {
830  // Clip to the container
831  v = gh->parent->x + gh->parent->width - ((const gcontainerVMT *)gh->parent->vmt)->RightBorder(gh->parent);
832  if (gh->x+w > v) w = v - gh->x;
833  v = gh->parent->y + gh->parent->height - ((const gcontainerVMT *)gh->parent->vmt)->BottomBorder(gh->parent);
834  if (gh->y+h > v) h = v - gh->y;
835  }
836  #endif
837 
838  // Clip to the screen
839  v = gdispGGetWidth(gh->display);
840  if (gh->x+w > v) w = v - gh->x;
841  v = gdispGGetHeight(gh->display);
842  if (gh->y+h > v) h = v - gh->y;
843 
844  // Give it a minimum size
845  if (w < MIN_WIN_WIDTH) w = MIN_WIN_WIDTH;
846  if (h < MIN_WIN_HEIGHT) h = MIN_WIN_HEIGHT;
847 
848  // If there has been no resize just exit
849  if (gh->width == w && gh->height == h)
850  return;
851 
852  // Set the new size and redraw
853  if ((gh->flags & GWIN_FLG_SYSVISIBLE)) {
854  if (w >= gh->width && h >= gh->height) {
855 
856  // The new size is larger - just redraw
857  gh->width = w; gh->height = h;
858  _gwinUpdate(gh);
859 
860  } else {
861  // We need to make this window invisible and ensure that has been drawn
862  gwinSetVisible(gh, FALSE);
863  _gwinFlushRedraws(REDRAW_WAIT);
864 
865  // Resize
866  gh->width = w; gh->height = h;
867 
868  #if GWIN_NEED_CONTAINERS
869  // Any children outside the new area need to be moved
870  if ((gh->flags & GWIN_FLG_CONTAINER)) {
871  GHandle child;
872 
873  // Move to their old relative location. THe WM_Move() will adjust as necessary
874  for(child = gwinGetFirstChild(gh); child; child = gwinGetSibling(child))
875  WM_Move(child, child->x-gh->x-((const gcontainerVMT *)gh->vmt)->LeftBorder(gh), child->y-gh->y-((const gcontainerVMT *)gh->vmt)->TopBorder(gh));
876  }
877  #endif
878 
879  // Mark it visible again in its new location
880  gwinSetVisible(gh, TRUE);
881  }
882  } else {
883  gh->width = w; gh->height = h;
884 
885  #if GWIN_NEED_CONTAINERS
886  // Any children outside the new area need to be moved
887  if ((gh->flags & GWIN_FLG_CONTAINER)) {
888  GHandle child;
889 
890  // Move to their old relative location. THe WM_Move() will adjust as necessary
891  for(child = gwinGetFirstChild(gh); child; child = gwinGetSibling(child))
892  WM_Move(child, child->x-gh->x-((const gcontainerVMT *)gh->vmt)->LeftBorder(gh), child->y-gh->y-((const gcontainerVMT *)gh->vmt)->TopBorder(gh));
893  }
894  #endif
895  }
896 }

898 static void WM_Move(GHandle gh, coord_t x, coord_t y) {
899  coord_t u, v;
900 
901  #if GWIN_NEED_CONTAINERS
902  if (gh->parent) {
903  // Clip to the parent size
904  u = gh->parent->width - ((const gcontainerVMT *)gh->parent->vmt)->LeftBorder(gh->parent) - ((const gcontainerVMT *)gh->parent->vmt)->RightBorder(gh->parent);
905  v = gh->parent->height - ((const gcontainerVMT *)gh->parent->vmt)->TopBorder(gh->parent) - ((const gcontainerVMT *)gh->parent->vmt)->BottomBorder(gh->parent);
906  } else
907  #endif
908  {
909  // Clip to the screen
910  u = gdispGGetWidth(gh->display);
911  v = gdispGGetHeight(gh->display);
912  }
913 
914  // Make sure we are positioned in the appropriate area
915  if (x+gh->width > u) x = u-gh->width;
916  if (x < 0) x = 0;
917  if (y+gh->height > v) y = v-gh->height;
918  if (y < 0) y = 0;
919 
920  // Make sure we don't overflow the appropriate area
921  u -= x;
922  v -= y;
923  if (gh->width < u) u = gh->width;
924  if (gh->height < v) v = gh->height;
925  if (u != gh->width || v != gh->height)
926  WM_Size(gh, u, v);
927 
928  #if GWIN_NEED_CONTAINERS
929  if (gh->parent) {
930  // Convert to a screen relative position
931  x += gh->parent->x + ((const gcontainerVMT *)gh->parent->vmt)->LeftBorder(gh->parent);
932  y += gh->parent->y + ((const gcontainerVMT *)gh->parent->vmt)->TopBorder(gh->parent);
933  }
934  #endif
935 
936  // If there has been no move just exit
937  if (gh->x == x && gh->y == y)
938  return;
939 
940  // Clear the old area and then redraw
941  if ((gh->flags & GWIN_FLG_SYSVISIBLE)) {
942  // We need to make this window invisible and ensure that has been drawn
943  gwinSetVisible(gh, FALSE);
944  _gwinFlushRedraws(REDRAW_WAIT);
945 
946  // Do the move
947  u = gh->x; gh->x = x;
948  v = gh->y; gh->y = y;
949 
950  #if GWIN_NEED_CONTAINERS
951  // Any children need to be moved
952  if ((gh->flags & GWIN_FLG_CONTAINER)) {
953  GHandle child;
954 
955  // Move to their old relative location. THe WM_Move() will adjust as necessary
956  for(child = gwinGetFirstChild(gh); child; child = gwinGetSibling(child))
957  WM_Move(child, child->x-u-((const gcontainerVMT *)gh->vmt)->LeftBorder(gh), child->y-v-((const gcontainerVMT *)gh->vmt)->TopBorder(gh));
958  }
959  #endif
960 
961  gwinSetVisible(gh, TRUE);
962  } else {
963  u = gh->x; gh->x = x;
964  v = gh->y; gh->y = y;
965 
966  #if GWIN_NEED_CONTAINERS
967  // Any children need to be moved
968  if ((gh->flags & GWIN_FLG_CONTAINER)) {
969  GHandle child;
970 
971  // Move to their old relative location. THe WM_Move() will adjust as necessary

984 static void WM_Raise(GHandle gh) {
985  // Take it off the list and then put it back on top
986  // The order of the list then reflects the z-order.
987 
988  gfxQueueASyncRemove(&_GWINList, &gh->wmq);
989  gfxQueueASyncPut(&_GWINList, &gh->wmq);
990 
991  #if GWIN_NEED_CONTAINERS
992  // Any children need to be raised too
993  if ((gh->flags & GWIN_FLG_CONTAINER)) {
994  GHandle gx;
995  GHandle child;
996  bool_t restart;
997 
998  // Raise the children too
999  // Note: Children can also have their own children so after each move we have to start again.
1000  for (gx = gwinGetNextWindow(0); gx; gx = gwinGetNextWindow(gx)) {
1001  if ((gx->flags & GWIN_FLG_CONTAINER)) {
1002  restart = FALSE;
1003  for (child = gwinGetNextWindow(0); child && child != gx; child = gwinGetNextWindow(child)) {
1004  if (child->parent == gx) {
1005  // Oops - this child is behind its parent. Move it to the front.
1006  gfxQueueASyncRemove(&_GWINList, &child->wmq);
1007  gfxQueueASyncPut(&_GWINList, &child->wmq);
1008 
1009  // Restart at the front of the list for this parent container as we have moved this child
1010  // to the end of the list. We also need to restart everything once this container is done.
1011  child = 0;
1012  restart = TRUE;
1013  }
1014  }
1015  if (restart)
1016  gx = 0;
1017  }
1018  }
1019  }
1020  #endif
1021 
1022  // Redraw the window
1023  _gwinUpdate(gh);
1024 }
