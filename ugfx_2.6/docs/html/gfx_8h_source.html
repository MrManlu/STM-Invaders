218  /**
219  * @brief The one call to start it all
220  *
221  * @note This will initialise each sub-system that has been turned on.
222  * For example, if GFX_USE_GDISP is defined then display will be initialised
223  * and cleared to black.
224  * @note If you define GFX_OS_NO_INIT as TRUE in your gfxconf.h file then ugfx doesn't try to
225  * initialise the operating system for you when you call @p gfxInit().
226  * @note If you define GFX_OS_EXTRA_INIT_FUNCTION in your gfxconf.h file the macro is the
227  * name of a void function with no parameters that is called immediately after
228  * operating system initialisation (whether or not GFX_OS_NO_INIT is set).
229  * @note If you define GFX_OS_EXTRA_DEINIT_FUNCTION in your gfxconf.h file the macro is the
230  * name of a void function with no parameters that is called immediately before
231  * operating system de-initialisation (as ugfx is exiting).
232  *
233  * @api
234  */
235  void gfxInit(void);
