110 /**
111  * Get ready for decoding a frame.
112  *
113  * Pre: Frame info has been read.
114  */
115 static gdispImageError startDecodeGif(gdispImage *img) {
116  gdispImagePrivate_GIF * priv;
117  gifimgdecode * decode;
118  uint16_t cnt;
119 
120  priv = (gdispImagePrivate_GIF *)img->priv;
121 
122  // We need the decode ram, and possibly a palette
123  if (!(decode = (gifimgdecode *)gdispImageAlloc(img, sizeof(gifimgdecode)+priv->frame.palsize*sizeof(color_t))))
124  return GDISP_IMAGE_ERR_NOMEMORY;
125 
126  // We currently have not read any image data block
127  decode->blocksz = 0;
128 
129  // Set the palette
130  if (priv->frame.palsize) {
131  // Local palette
132  decode->maxpixel = priv->frame.palsize-1;
133  decode->palette = (color_t *)(decode+1);
134  gfileSetPos(img->f, priv->frame.pospal);
135  for(cnt = 0; cnt < priv->frame.palsize; cnt++) {
136  if (gfileRead(img->f, &decode->buf, 3) != 3)
137  goto baddatacleanup;
138  decode->palette[cnt] = RGB2COLOR(decode->buf[0], decode->buf[1], decode->buf[2]);
139  }
140  } else if (priv->palette) {
141  // Global palette
142  decode->maxpixel = priv->palsize-1;
143  decode->palette = priv->palette;
144  } else {
145  // Oops - we must have a palette
146  goto baddatacleanup;
147  }
148 
149  // Get the initial lzw code size and values
150  gfileSetPos(img->f, priv->frame.posimg);
151  if (gfileRead(img->f, &decode->bitsperpixel, 1) != 1 || decode->bitsperpixel >= GIF_MAX_CODE_BITS)
152  goto baddatacleanup;
153  decode->code_clear = 1 << decode->bitsperpixel;
154  decode->code_eof = decode->code_clear + 1;
155  decode->code_max = decode->code_clear + 2;
156  decode->code_last = GIF_CODE_NONE;
157  decode->bitspercode = decode->bitsperpixel+1;
158  decode->maxcodesz = 1 << decode->bitspercode;
159  decode->shiftbits = 0;
160  decode->shiftdata = 0;
161  decode->stackcnt = 0;
162  for(cnt = 0; cnt <= GIF_CODE_MAX; cnt++)
163  decode->prefix[cnt] = GIF_CODE_NONE;
164 
165  // All ready to go
166  priv->decode = decode;
167  return GDISP_IMAGE_ERR_OK;
168 
169 baddatacleanup:
170  gdispImageFree(img, decode, sizeof(gifimgdecode)+priv->frame.palsize*sizeof(color_t));
171  return GDISP_IMAGE_ERR_BADDATA;
172 }

191 static uint16_t getPrefixGif(gifimgdecode *decode, uint16_t code) {
192  uint16_t i;
193 
194  for(i=0; code > decode->code_clear && i <= GIF_CODE_MAX; i++, code = decode->prefix[code]) {
195  if (code > GIF_CODE_MAX)
196  return GIF_CODE_NONE;
197  }
198  return code;

201 /**
202  * Decode some pixels from a frame.
203  *
204  * Pre: We are ready for decoding.
205  *
206  * Return: The number of pixels decoded 0 .. BLIT_BUFFER_SIZE_GIF-1. 0 means EOF
207  *
208  * Note: The resulting pixels are stored in decode->buf
209  */
210 static uint16_t getBytesGif(gdispImage *img) {
211  gdispImagePrivate_GIF * priv;
212  gifimgdecode * decode;
213  uint16_t cnt;
214  uint16_t code, prefix;
215  uint8_t bdata;
216 
217  priv = (gdispImagePrivate_GIF *)img->priv;
218  decode = priv->decode;
219  cnt = 0;
220 
221  // At EOF
222  if (decode->code_last == decode->code_eof)
223  return 0;
224 
225  while(cnt < sizeof(decode->buf)) {
226  // Use the stack up first
227  if (decode->stackcnt > 0) {
228  decode->buf[cnt++] = decode->stack[--decode->stackcnt];
229  continue;
230  }
231 
232  // Get another code - a code is made up of decode->bitspercode bits.
233  while (decode->shiftbits < decode->bitspercode) {
234  // Get a byte - we may have to start a new data block
235  if ((!decode->blocksz && (gfileRead(img->f, &decode->blocksz, 1) != 1 || !decode->blocksz))
236  || gfileRead(img->f, &bdata, 1) != 1) {
237  // Pretend we got the EOF code - some encoders seem to just end the file
238  decode->code_last = decode->code_eof;
239  return cnt;
240  }
241  decode->blocksz--;
242 
243  decode->shiftdata |= ((unsigned long)bdata) << decode->shiftbits;
244  decode->shiftbits += 8;
245  }
246  code = decode->shiftdata & GifBitMask[decode->bitspercode];
247  decode->shiftdata >>= decode->bitspercode;
248  decode->shiftbits -= decode->bitspercode;
249  /**
250  * If code cannot fit into bitspercode bits we must raise its size.
251  * Note that codes above GIF_CODE_MAX are used for special signaling.
252  * If we're using GIF_MAX_CODE_BITS bits already and we're at the max code, just
253  * keep using the table as it is, don't increment decode->bitspercode.
254  */
255  if (decode->code_max < GIF_CODE_MAX + 2 && ++decode->code_max > decode->maxcodesz && decode->bitspercode < GIF_MAX_CODE_BITS) {
256  decode->maxcodesz <<= 1;
257  decode->bitspercode++;
258  }
259 
260  // EOF - the appropriate way to stop decoding
261  if (code == decode->code_eof) {
262  // Skip to the end of the data blocks
263  do {
264  gfileSetPos(img->f, gfileGetPos(img->f)+decode->blocksz);
265  } while (gfileRead(img->f, &decode->blocksz, 1) == 1 && decode->blocksz);
266 
267  // Mark the end
268  decode->code_last = decode->code_eof;
269  break;
270  }
271 
272  if (code == decode->code_clear) {
273  // Start again
274  for(prefix = 0; prefix <= GIF_CODE_MAX; prefix++)
275  decode->prefix[prefix] = GIF_CODE_NONE;
276  decode->code_max = decode->code_eof + 1;
277  decode->bitspercode = decode->bitsperpixel + 1;
278  decode->maxcodesz = 1 << decode->bitspercode;
279  decode->code_last = GIF_CODE_NONE;
280  continue;
281  }
282 
283  if (code < decode->code_clear) {
284  // Simple unencoded pixel - add it
285  decode->buf[cnt++] = code;
286 
287  } else {
288  /**
289  * Its a LZW code - trace the linked list until the prefix is a
290  * valid pixel while pushing the suffix pixels on the stack.
291  * If done, pop the stack in reverse order adding the pixels
292  */
293  if (decode->prefix[code] != GIF_CODE_NONE)
294  prefix = code;
295 
296  /**
297  * Only allowed if the code equals the partial code.
298  * In that case code = XXXCode, CrntCode or the
299  * prefix code is last code and the suffix char is
300  * exactly the prefix of last code!
301  */
302  else if (code == decode->code_max - 2 && decode->stackcnt < sizeof(decode->stack)) {
303  prefix = decode->code_last;
304  decode->suffix[decode->code_max - 2] = decode->stack[decode->stackcnt++] = getPrefixGif(decode, decode->code_last);
305  } else
306  return 0;
307 
308  /**
309  * If the image is OK we should not get a GIF_CODE_NONE while tracing.
310  * To prevent looping with a bad image we use StackPtr as loop counter
311  * and stop before overflowing Stack[].
312  */
313  while (decode->stackcnt < sizeof(decode->stack) && prefix > decode->code_clear && prefix <= GIF_CODE_MAX) {
314  decode->stack[decode->stackcnt++] = decode->suffix[prefix];
315  prefix = decode->prefix[prefix];
316  }
317  if (decode->stackcnt >= sizeof(decode->stack) || prefix > GIF_CODE_MAX)
318  return 0;
319 
320  /* Push the last character on stack: */
321  decode->stack[decode->stackcnt++] = prefix;
322  }
323 
324  if (decode->code_last != GIF_CODE_NONE && decode->prefix[decode->code_max - 2] == GIF_CODE_NONE) {
325  decode->prefix[decode->code_max - 2] = decode->code_last;
326 
327  /* Only allowed if code is exactly the running code:
328  * In that case code = XXXCode, CrntCode or the
329  * prefix code is last code and the suffix char is
330  * exactly the prefix of last code! */
331  decode->suffix[decode->code_max - 2] = getPrefixGif(decode, code == decode->code_max - 2 ? decode->code_last : code);
332  }
333  decode->code_last = code;
334  }
335  return cnt;
336 }

339  * Read the info on a frame.
340  *
341  * Pre: The file position is at the start of the frame.
342  */
343 static gdispImageError initFrameGif(gdispImage *img) {
344  gdispImagePrivate_GIF * priv;
345  gifimgcache * cache;
346  uint8_t blocktype;
347  uint8_t blocksz;
348 
349  priv = (gdispImagePrivate_GIF *)img->priv;
350 
351  // Save the dispose info from the existing frame
352  priv->dispose.flags = priv->frame.flags;
353  priv->dispose.paltrans = priv->frame.paltrans;
354  priv->dispose.x = priv->frame.x;
355  priv->dispose.y = priv->frame.y;
356  priv->dispose.width = priv->frame.width;
357  priv->dispose.height = priv->frame.height;
358 
359  // Check for a cached version of this image
360  for(cache=priv->cache; cache && cache->frame.posstart <= (size_t)gfileGetPos(img->f); cache=cache->next) {
361  if (cache->frame.posstart == (size_t)gfileGetPos(img->f)) {
362  priv->frame = cache->frame;
363  priv->curcache = cache;
364  return GDISP_IMAGE_ERR_OK;
365  }
366  }
367 
368  // Get ready for a new image
369  priv->curcache = 0;
370  priv->frame.posstart = gfileGetPos(img->f);
371  priv->frame.flags = 0;
372  priv->frame.delay = 0;
373  priv->frame.palsize = 0;
374 
375  // Process blocks until we reach the image descriptor
376  while(1) {
377  if (gfileRead(img->f, &blocktype, 1) != 1)
378  return GDISP_IMAGE_ERR_BADDATA;
379 
380  switch(blocktype) {
381  case 0x2C: //',' - IMAGE_DESC_RECORD_TYPE;
382  // Read the Image Descriptor
383  if (gfileRead(img->f, priv->buf, 9) != 9)
384  return GDISP_IMAGE_ERR_BADDATA;
385  priv->frame.x = gdispImageGetAlignedLE16(priv->buf, 0);
386  priv->frame.y = gdispImageGetAlignedLE16(priv->buf, 2);
387  priv->frame.width = gdispImageGetAlignedLE16(priv->buf, 4);
388  priv->frame.height = gdispImageGetAlignedLE16(priv->buf, 6);
389  if (((uint8_t *)priv->buf)[8] & 0x80) // Local color table?
390  priv->frame.palsize = 2 << (((uint8_t *)priv->buf)[8] & 0x07);
391  if (((uint8_t *)priv->buf)[8] & 0x40) // Interlaced?
392  priv->frame.flags |= GIFL_INTERLACE;
393 
394  // We are ready to go for the actual palette read and image decode
395  priv->frame.pospal = gfileGetPos(img->f);
396  priv->frame.posimg = priv->frame.pospal+priv->frame.palsize*3;
397  priv->frame.posend = 0;
398 
399  // Mark this as an animated image if more than 1 frame.
400  if (priv->frame.posstart != priv->frame0pos)
401  img->flags |= GDISP_IMAGE_FLG_ANIMATED;
402  return GDISP_IMAGE_ERR_OK;
403 
404  case 0x21: //'!' - EXTENSION_RECORD_TYPE;
405  // Read the extension type
406  if (gfileRead(img->f, &blocktype, 1) != 1)
407  return GDISP_IMAGE_ERR_BADDATA;
408 
409  switch(blocktype) {
410  case 0xF9: // EXTENSION - Graphics Control Block
411  // Read the GCB
412  if (gfileRead(img->f, priv->buf, 6) != 6)
413  return GDISP_IMAGE_ERR_BADDATA;
414  // Check we have read a 4 byte data block and a data block terminator (0)
415  if (((uint8_t *)priv->buf)[0] != 4 || ((uint8_t *)priv->buf)[5] != 0)
416  return GDISP_IMAGE_ERR_BADDATA;
417  // Process the flags
418  switch(((uint8_t *)priv->buf)[1] & 0x1C) {
419  case 0x00: case 0x04: break; // Dispose = do nothing
420  case 0x08: priv->frame.flags |= GIFL_DISPOSECLEAR; break; // Dispose = clear
421  case 0x0C: case 0x10: priv->frame.flags |= GIFL_DISPOSEREST; break; // Dispose = restore. Value 0x10 is a hack for bad encoders
422  default: return GDISP_IMAGE_ERR_UNSUPPORTED;
423  }
424  if (((uint8_t *)priv->buf)[1] & 0x01) {
425  priv->frame.flags |= GIFL_TRANSPARENT;
426  img->flags |= GDISP_IMAGE_FLG_TRANSPARENT; // We set this but never clear it
427  }
428  if (((uint8_t *)priv->buf)[1] & 0x02) // Wait for user input?
429  img->flags |= GDISP_IMAGE_FLG_MULTIPAGE;
430  else
431  img->flags &= ~GDISP_IMAGE_FLG_MULTIPAGE;
432  // Process frame delay and the transparent color (if any)
433  priv->frame.delay = gdispImageGetAlignedLE16(priv->buf, 2);
434  priv->frame.paltrans = ((uint8_t *)priv->buf)[4];
435  break;
436 
437  case 0xFF: // EXTENSION - Application
438  // We only handle this for the special Netscape loop counter for animation
439  if (priv->flags & GIF_LOOP)
440  goto skipdatablocks;
441  // Read the Application header
442  if (gfileRead(img->f, priv->buf, 16) != 16)
443  return GDISP_IMAGE_ERR_BADDATA;
444  // Check we have read a 11 byte data block
445  if (((uint8_t *)priv->buf)[0] != 11 && ((uint8_t *)priv->buf)[12] != 3)
446  return GDISP_IMAGE_ERR_BADDATA;
447  // Check the vendor
448  if (((uint8_t *)priv->buf)[1] == 'N' && ((uint8_t *)priv->buf)[2] == 'E' && ((uint8_t *)priv->buf)[3] == 'T'
449  && ((uint8_t *)priv->buf)[4] == 'S' && ((uint8_t *)priv->buf)[5] == 'C' && ((uint8_t *)priv->buf)[6] == 'A'
450  && ((uint8_t *)priv->buf)[7] == 'P' && ((uint8_t *)priv->buf)[8] == 'E' && ((uint8_t *)priv->buf)[9] == '2'
451  && ((uint8_t *)priv->buf)[10] == '.' && ((uint8_t *)priv->buf)[11] == '0') {
452  if (((uint8_t *)priv->buf)[13] == 1) {
453  priv->loops = gdispImageGetAlignedLE16(priv->buf, 14);
454  priv->flags |= GIF_LOOP;
455  if (!priv->loops)
456  priv->flags |= GIF_LOOPFOREVER;
457  }
458  }
459  goto skipdatablocks;
460 
461  case 0x01: // EXTENSION - Plain Text (Graphics Rendering)
462  case 0xFE: // EXTENSION - Comment
463  default:
464  // 0x00-0x7F (0-127) are the Graphic Rendering blocks
465  if (blocktype <= 0x7F)
466  return GDISP_IMAGE_ERR_UNSUPPORTED;
467  // 0x80-0xF9 (128-249) are the Control blocks
468  // 0xFA-0xFF (250-255) are the Special Purpose blocks
469  // We don't understand this extension - just skip it by skipping data blocks
470  skipdatablocks:
471  while(1) {
472  if (gfileRead(img->f, &blocksz, 1) != 1)
473  return GDISP_IMAGE_ERR_BADDATA;
474  if (!blocksz)
475  break;
476  gfileSetPos(img->f, gfileGetPos(img->f) + blocksz);
477  }
478  break;
479  }
480  break;
481 
482  case 0x3B: //';' - TERMINATE_RECORD_TYPE;
483  // Are we an looping animation
484  if (!(priv->flags & GIF_LOOP))
485  return GDISP_IMAGE_GIF_EOF;
486  if (!(priv->flags & GIF_LOOPFOREVER)) {
487  if (!priv->loops)
488  return GDISP_IMAGE_GIF_EOF;
489  priv->loops--;
490  }
491 
492  // Seek back to frame0
493  gfileSetPos(img->f, priv->frame0pos);
494  return GDISP_IMAGE_GIF_LOOP;
495 
496  default: // UNDEFINED_RECORD_TYPE;
497  return GDISP_IMAGE_ERR_UNSUPPORTED;
498  }
499  }
500 }

503  gdispImagePrivate_GIF * priv;
504  gifimgcache * cache;
505  gifimgcache * ncache;
506 
507  priv = (gdispImagePrivate_GIF *)img->priv;
508  if (priv) {
509  // Free any stored frames

511  while(cache) {
512  ncache = cache->next;

520  }

523 gdispImageError gdispImageOpen_GIF(gdispImage *img) {
524  gdispImagePrivate_GIF *priv;
525  uint8_t hdr[6];
526  uint16_t aword;
527 
528  /* Read the file identifier */
529  if (gfileRead(img->f, hdr, 6) != 6)
530  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
531 
532  /* Process the GIFFILEHEADER structure */
533 
534  if (hdr[0] != 'G' || hdr[1] != 'I' || hdr[2] != 'F'
535  || hdr[3] != '8' || (hdr[4] != '7' && hdr[4] != '9') || hdr[5] != 'a')
536  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
537 
538  /* We know we are a GIF format image */
539  img->flags = 0;
540 
541  /* Allocate our private area */
542  if (!(img->priv = gdispImageAlloc(img, sizeof(gdispImagePrivate_GIF))))
543  return GDISP_IMAGE_ERR_NOMEMORY;
544 
545  /* Initialise the essential bits in the private area */
546  priv = (gdispImagePrivate_GIF *)img->priv;
547  priv->flags = 0;
548  priv->palsize = 0;
549  priv->palette = 0;
550  priv->frame.flags = 0;
551  priv->cache = 0;
552  priv->curcache = 0;
553  priv->decode = 0;
554 
555  /* Process the Screen Descriptor structure */
556 
557  // Read the screen descriptor
558  if (gfileRead(img->f, priv->buf, 7) != 7)
559  goto baddatacleanup;
560  // Get the width
561  img->width = gdispImageGetAlignedLE16(priv->buf, 0);
562  // Get the height
563  img->height = gdispImageGetAlignedLE16(priv->buf, 2);
564  if (((uint8_t *)priv->buf)[4] & 0x80) {
565  // Global color table
566  priv->palsize = 2 << (((uint8_t *)priv->buf)[4] & 0x07);
567  // Allocate the global palette

588  gdispImageClose_GIF(img); // Clean up the private data area

590  case GDISP_IMAGE_ERR_NOMEMORY: // Out of Memory
591  nomemcleanup:

595  case GDISP_IMAGE_GIF_LOOP: // We should have a frame but we don't seem to
596  case GDISP_IMAGE_ERR_BADDATA: // Oops - something wrong with the data
597  default:

604 gdispImageError gdispImageCache_GIF(gdispImage *img) {

607  gifimgdecode * decode;
608  uint8_t * p;
609  uint8_t * q;
610  coord_t mx, my;
611  uint16_t cnt;
612 
613  /* If we are already cached - just return OK */
614  priv = (gdispImagePrivate_GIF *)img->priv;
615  if (priv->curcache)
616  return GDISP_IMAGE_ERR_OK;
617

619  if (!(cache = (gifimgcache *)gdispImageAlloc(img, sizeof(gifimgcache) + priv->frame.palsize*sizeof(color_t) + priv->frame.width*priv->frame.height)))
620  return GDISP_IMAGE_ERR_NOMEMORY;
621

623  decode = 0;
624  cache->frame = priv->frame;
625  cache->imagebits = (uint8_t *)(cache+1) + cache->frame.palsize*sizeof(color_t);
626  cache->next = 0;
627 
628  /* Start the decode */
629  switch(startDecodeGif(img)) {
630  case GDISP_IMAGE_ERR_OK: break;
631  case GDISP_IMAGE_ERR_NOMEMORY: goto nomemcleanup;
632  case GDISP_IMAGE_ERR_BADDATA:
633  default: goto baddatacleanup;
634  }

636 
637  // Save the palette
638  if (cache->frame.palsize) {
639  cache->palette = (color_t *)(cache+1);
640 
641  /* Copy the local palette into the cache */
642  for(cnt = 0; cnt < cache->frame.palsize; cnt++)
643  cache->palette[cnt] = decode->palette[cnt];
644  } else
645  cache->palette = priv->palette;
646 
647  // Check for interlacing
648  cnt = 0;
649  q = 0;
650  if (cache->frame.flags & GIFL_INTERLACE) {
651  // Every 8th row starting at row 0
652  for(p=cache->imagebits, my=0; my < cache->frame.height; my+=8, p += cache->frame.width*7) {
653  for(mx=0; mx < cache->frame.width; mx++) {
654  if (!cnt) {
655  if (!(cnt = getBytesGif(img))) {
656  // Sometimes the image EOF is a bit early - treat the rest as transparent
657  if (decode->code_last != decode->code_eof)
658  goto baddatacleanup;
659  while(cnt < sizeof(decode->buf))
660  decode->buf[cnt++] = (cache->frame.flags & GIFL_TRANSPARENT) ? cache->frame.paltrans : 0;
661  }
662  q = decode->buf;
663  }
664  *p++ = *q++;
665  cnt--;
666  }
667  }
668  // Every 8th row starting at row 4
669  for(p=cache->imagebits+cache->frame.width*4, my=4; my < cache->frame.height; my+=8, p += cache->frame.width*7) {
670  for(mx=0; mx < cache->frame.width; mx++) {
671  if (!cnt) {
672  if (!(cnt = getBytesGif(img))) {
673  // Sometimes the image EOF is a bit early - treat the rest as transparent
674  if (decode->code_last != decode->code_eof)
675  goto baddatacleanup;

678  }
679  q = decode->buf;
680  }
681  *p++ = *q++;
682  cnt--;
683  }
684  }
685  // Every 4th row starting at row 2
686  for(p=cache->imagebits+cache->frame.width*2, my=2; my < cache->frame.height; my+=4, p += cache->frame.width*3) {
687  for(mx=0; mx < cache->frame.width; mx++) {
688  if (!cnt) {
689  if (!(cnt = getBytesGif(img))) {
690  // Sometimes the image EOF is a bit early - treat the rest as transparent
691  if (decode->code_last != decode->code_eof)
692  goto baddatacleanup;
693  while(cnt < sizeof(decode->buf))
694  decode->buf[cnt++] = (cache->frame.flags & GIFL_TRANSPARENT) ? cache->frame.paltrans : 0;
695  }
696  q = decode->buf;
697  }
698  *p++ = *q++;
699  cnt--;
700  }
701  }
702  // Every 2nd row starting at row 1
703  for(p=cache->imagebits+cache->frame.width, my=1; my < cache->frame.height; my+=2, p += cache->frame.width) {
704  for(mx=0; mx < cache->frame.width; mx++) {
705  if (!cnt) {
706  if (!(cnt = getBytesGif(img))) {
707  // Sometimes the image EOF is a bit early - treat the rest as transparent
708  if (decode->code_last != decode->code_eof)
709  goto baddatacleanup;
710  while(cnt < sizeof(decode->buf))
711  decode->buf[cnt++] = (cache->frame.flags & GIFL_TRANSPARENT) ? cache->frame.paltrans : 0;
712  }
713  q = decode->buf;
714  }
715  *p++ = *q++;
716  cnt--;
717  }
718  }
719  } else {
720  // Every row in sequence
721  p=cache->imagebits;
722  for(my=0; my < cache->frame.height; my++) {
723  for(mx=0; mx < cache->frame.width; mx++) {
724  if (!cnt) {
725  if (!(cnt = getBytesGif(img))) {
726  // Sometimes the image EOF is a bit early - treat the rest as transparent
727  if (decode->code_last != decode->code_eof)
728  goto baddatacleanup;
729  while(cnt < sizeof(decode->buf))
730  decode->buf[cnt++] = (cache->frame.flags & GIFL_TRANSPARENT) ? cache->frame.paltrans : 0;
731  }
732  q = decode->buf;
733  }
734  *p++ = *q++;
735  cnt--;
736  }
737  }
738  }
739  // We could be pedantic here but extra bytes won't hurt us
740  while(getBytesGif(img));
741  priv->frame.posend = cache->frame.posend = gfileGetPos(img->f);
742 
743  // Save everything
744  priv->curcache = cache;
745  if (!priv->cache)
746  priv->cache = cache;
747  else if (priv->cache->frame.posstart > cache->frame.posstart) {
748  cache->next = priv->cache;
749  priv->cache = cache;
750  } else {
751  gifimgcache *pc;
752 
753  for(pc = priv->cache; pc; pc = pc->next) {
754  if (!pc->next || pc->next->frame.posstart > cache->frame.posstart) {
755  cache->next = pc->next;
756  pc->next = cache;
757  break;
758  }
759  }
760  }
761  stopDecodeGif(img);
762  return GDISP_IMAGE_ERR_OK;
763 
764 nomemcleanup:
765  stopDecodeGif(img);
766  gdispImageFree(img, cache, sizeof(gifimgcache) + priv->frame.palsize*sizeof(color_t) + priv->frame.width*priv->frame.height);
767  return GDISP_IMAGE_ERR_NOMEMORY;
768 
769 baddatacleanup:
770  stopDecodeGif(img);
771  gdispImageFree(img, cache, sizeof(gifimgcache) + priv->frame.palsize*sizeof(color_t) + priv->frame.width*priv->frame.height);
772  return GDISP_IMAGE_ERR_BADDATA;
773 }
