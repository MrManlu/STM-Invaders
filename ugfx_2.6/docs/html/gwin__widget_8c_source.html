96 static void gwidgetEvent(void *param, GEvent *pe) {
97  #define pme ((GEventMouse *)pe)
98  #define pke ((GEventKeyboard *)pe)
99  #define pte ((GEventToggle *)pe)
100  #define pde ((GEventDial *)pe)
101 
102  GHandle h;
103  GHandle gh;
104  #if GFX_USE_GINPUT && (GINPUT_NEED_TOGGLE || GINPUT_NEED_DIAL)
105  uint16_t role;
106  #endif
107  (void) param;
108 
109  // Process various events
110  switch (pe->type) {
111 
112  #if GFX_USE_GINPUT && GINPUT_NEED_MOUSE
113  case GEVENT_MOUSE:
114  case GEVENT_TOUCH:
115  // Cycle through all windows
116  for (gh = 0, h = gwinGetNextWindow(0); h; h = gwinGetNextWindow(h)) {
117 
118  // The window must be on this display and visible to be relevant
119  if (h->display != pme->display || !(h->flags & GWIN_FLG_SYSVISIBLE))
120  continue;
121 
122  // Is the mouse currently captured by this widget?
123  if ((h->flags & (GWIN_FLG_WIDGET|GWIN_FLG_MOUSECAPTURE)) == (GWIN_FLG_WIDGET|GWIN_FLG_MOUSECAPTURE)) {
124  gh = h;
125  if ((pme->buttons & GMETA_MOUSE_UP)) {
126  gh->flags &= ~GWIN_FLG_MOUSECAPTURE;
127  if (wvmt->MouseUp)
128  wvmt->MouseUp(gw, pme->x - gh->x, pme->y - gh->y);
129  } else if (wvmt->MouseMove)
130  wvmt->MouseMove(gw, pme->x - gh->x, pme->y - gh->y);
131 
132  // There is only ever one captured mouse. Prevent normal mouse processing if there is a captured mouse
133  gh = 0;
134 
135  break;
136  }
137 
138  // Save the highest z-order window that the mouse is over
139  if (pme->x >= h->x && pme->x < h->x + h->width && pme->y >= h->y && pme->y < h->y + h->height)
140  gh = h;
141  }
142 
143  // Process any mouse down over the highest order window if it is an enabled widget
144  if (gh && (gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED)) == (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED)) {
145  if ((pme->buttons & GMETA_MOUSE_DOWN)) {
146  gh->flags |= GWIN_FLG_MOUSECAPTURE;
147 
148  #if (GFX_USE_GINPUT && GINPUT_NEED_KEYBOARD) || GWIN_NEED_KEYBOARD
149  // We should try and capture the focus on this window.
150  // If we can't then we don't change the focus
151  gwinSetFocus(gh);
152  #endif
153 
154  if (wvmt->MouseDown)
155  wvmt->MouseDown(gw, pme->x - gh->x, pme->y - gh->y);
156  }
157  }
158  break;
159  #endif
160 
161  #if (GFX_USE_GINPUT && GINPUT_NEED_KEYBOARD) || GWIN_NEED_KEYBOARD
162  case GEVENT_KEYBOARD:
163  // If Tab key pressed then set focus to next widget
164  if (pke->bytecount == 1 && pke->c[0] == GKEY_TAB) {
165  if (!(pke->keystate & GKEYSTATE_KEYUP))
166  _gwinMoveFocus();
167  break;
168  }
169 
170  // Otherwise, send keyboard events only to widget in focus
171  if (_widgetInFocus)
172  ((gwidgetVMT*)_widgetInFocus->vmt)->KeyboardEvent((GWidgetObject*)_widgetInFocus, pke);
173  break;
174  #endif
175 
176  #if GFX_USE_GINPUT && GINPUT_NEED_TOGGLE
177  case GEVENT_TOGGLE:
178  // Cycle through all windows
179  for(gh = gwinGetNextWindow(0); gh; gh = gwinGetNextWindow(gh)) {
180 
181  // check if it a widget that is enabled and visible
182  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED|GWIN_FLG_SYSVISIBLE)) != (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED|GWIN_FLG_SYSVISIBLE))
183  continue;
184 
185  for(role = 0; role < wvmt->toggleroles; role++) {
186  if (wvmt->ToggleGet(gw, role) == pte->instance) {
187  if (pte->on) {
188  if (wvmt->ToggleOn)
189  wvmt->ToggleOn(gw, role);
190  } else {
191  if (wvmt->ToggleOff)
192  wvmt->ToggleOff(gw, role);
193  }
194  }
195  }
196  }
197  break;
198  #endif
199 
200  #if GFX_USE_GINPUT && GINPUT_NEED_DIAL
201  case GEVENT_DIAL:
202  // Cycle through all windows
203  for(gh = gwinGetNextWindow(0); gh; gh = gwinGetNextWindow(gh)) {
204 
205  // check if it a widget that is enabled and visible
206  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED|GWIN_FLG_SYSVISIBLE)) != (GWIN_FLG_WIDGET|GWIN_FLG_SYSENABLED|GWIN_FLG_SYSVISIBLE))
207  continue;
208 
209  for(role = 0; role < wvmt->dialroles; role++) {
210  if (wvmt->DialGet(gw, role) == pte->instance) {
211  if (wvmt->DialMove)
212  wvmt->DialMove(gw, role, pde->value, pde->maxvalue);
213  }
214  }
215  }
216  break;
217  #endif
218 
219  default:
220  break;
221  }
222 
223  #undef pme
224  #undef pte
225  #undef pke
226  #undef pde
227 }

234  bool_t gwinSetFocus(GHandle gh) {
235  GHandle oldFocus;
236 
237  // Do we already have the focus?
238  if (gh == _widgetInFocus)
239  return TRUE;
240 
241  // The new window must be NULLL or a visible enabled widget with a keyboard handler
242  if (!gh || ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)) == (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)
243  && ((gwidgetVMT*)gh->vmt)->KeyboardEvent)) {
244  // Move the current focus
245  oldFocus = _widgetInFocus;
246  _widgetInFocus = gh;
247  if (oldFocus) _gwinUpdate(oldFocus);
248  if (gh) _gwinUpdate(gh);
249  return TRUE;
250  }
251  return FALSE;
252  }

254  void _gwinMoveFocus(void) {
255  GHandle gh;
256  bool_t looponce;
257 
258  // Find a new focus window (one may or may not exist).
259  looponce = FALSE;
260  for(gh = gwinGetNextWindow(_widgetInFocus); ; gh = gwinGetNextWindow(gh)) {
261  if (!gh && !looponce) {
262  looponce = TRUE;
263  gh = gwinGetNextWindow(0);
264  }
265  if (gwinSetFocus(gh))
266  break;
267  }
268  }

270  void _gwinFixFocus(GHandle gh) {
271  GHandle oldFocus;
272 
273  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)) == (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)
274  && ((gwidgetVMT*)gh->vmt)->KeyboardEvent) {
275 
276  // We are a candidate to be able to claim the focus
277 
278  // Claim the focus if no-one else has
279  if (!_widgetInFocus)
280  _widgetInFocus = gh;
281 
282  return;
283  }
284 
285  // We have lost any right to the focus
286 
287  // Did we have the focus
288  if (gh != _widgetInFocus)
289  return;
290 
291  // We did - we need to find a new focus window
292  oldFocus = _widgetInFocus;
293  for(gh = gwinGetNextWindow(oldFocus); gh && gh != oldFocus; gh = gwinGetNextWindow(gh)) {
294 
295  // Must be a visible enabled widget with a keyboard handler
296  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)) == (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_SYSENABLED|GWIN_FLG_VISIBLE|GWIN_FLG_SYSVISIBLE)
297  && ((gwidgetVMT*)gh->vmt)->KeyboardEvent) {
298 
299  // Grab the focus for the new window
300  _widgetInFocus = gh;
301 
302  // This new window still needs to be marked for redraw (but don't actually do it yet).
303  gh->flags |= GWIN_FLG_NEEDREDRAW;
304  // RedrawPending |= DOREDRAW_VISIBLES; - FIX LATER
305  return;
306  }
307  }
308 
309  // No-one has the right to the focus
310  _widgetInFocus = 0;
311  }

401 void _gwidgetDestroy(GHandle gh) {
402  #if GFX_USE_GINPUT && (GINPUT_NEED_TOGGLE || GINPUT_NEED_DIAL)
403  uint16_t role, instance;
404  #endif
405 
406  // Make the window is invisible so it is not eligible for focus
407  gh->flags &= ~GWIN_FLG_VISIBLE;
408  _gwinFixFocus(gh);
409 
410  // Deallocate the text (if necessary)
411  if ((gh->flags & GWIN_FLG_ALLOCTXT)) {
412  gh->flags &= ~GWIN_FLG_ALLOCTXT;
413  gfxFree((void *)gw->text);
414  }
415 
416  #if GFX_USE_GINPUT && GINPUT_NEED_TOGGLE
417  // Detach any toggles from this object
418  for(role = 0; role < wvmt->toggleroles; role++) {
419  instance = wvmt->ToggleGet(gw, role);
420  if (instance != GWIDGET_NO_INSTANCE) {
421  wvmt->ToggleAssign(gw, role, GWIDGET_NO_INSTANCE);
422  if (!FindToggleUser(instance))
423  geventDetachSource(&gl, ginputGetToggle(instance));
424  }
425  }
426  #endif
427 
428  #if GFX_USE_GINPUT && GINPUT_NEED_DIAL
429  // Detach any dials from this object
430  for(role = 0; role < wvmt->dialroles; role++) {
431  instance = wvmt->DialGet(gw, role);
432  if (instance != GWIDGET_NO_INSTANCE) {
433  wvmt->DialAssign(gw, role, GWIDGET_NO_INSTANCE);
434  if (!FindDialUser(instance))
435  geventDetachSource(&gl, ginputGetDial(instance));

437  }
438  #endif
439 
440  // Remove any listeners on this object.
441  geventDetachSourceListeners((GSourceHandle)gh);

626  bool_t gwinAttachToggle(GHandle gh, uint16_t role, uint16_t instance) {
627  GSourceHandle gsh;
628  uint16_t oi;
629 
630  // Is this a widget
631  if (!(gh->flags & GWIN_FLG_WIDGET))
632  return FALSE;
633 
634  // Is the role valid
635  if (role >= wvmt->toggleroles)
636  return FALSE;
637

639  if (!(gsh = ginputGetToggle(instance)))

641
