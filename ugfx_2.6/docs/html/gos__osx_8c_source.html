91 gfxThreadHandle gfxThreadCreate(void *stackarea, size_t stacksz, threadpriority_t prio, DECLARE_THREAD_FUNCTION((*fn),p), void *param) {
92  gfxThreadHandle th;
93  (void) stackarea;
94  (void) stacksz;
95  (void) prio;
96 
97  // Implementing priority with pthreads is a rats nest that is also pthreads implementation dependent.
98  // Only some pthreads schedulers support it, some implementations use the operating system process priority mechanisms.
99  // Even those that do support it can have different ranges of priority and "normal" priority is an undefined concept.
100  // Across different UNIX style operating systems things can be very different (let alone OS's such as Windows).
101  // Even just Linux changes the way priority works with different kernel schedulers and across kernel versions.
102  // For these reasons we ignore the priority.
103 
104  if (pthread_create(&th, 0, fn, param))
105  return 0;
106  return th;
107 }

131 bool_t gfxSemWait(gfxSem *pSem, delaytime_t ms) {
132  pthread_mutex_lock(&pSem->mtx);
133  switch (ms) {
134  case TIME_INFINITE:
135  while (!pSem->cnt)
136  pthread_cond_wait(&pSem->cond, &pSem->mtx);
137  break;
138  case TIME_IMMEDIATE:
139  if (!pSem->cnt) {
140  pthread_mutex_unlock(&pSem->mtx);
141  return FALSE;
142  }
143  break;
144  default:
145  {
146  struct timeval now;
147  struct timespec tm;
148 
149  gettimeofday(&now, 0);
150  tm.tv_sec = now.tv_sec + ms / 1000;
151  tm.tv_nsec = now.tv_usec * 1000 + (ms % 1000) * 1000000;
152  while (!pSem->cnt) {
153  // We used to test the return value for ETIMEDOUT. This doesn't
154  // work in some current pthread libraries which return -1 instead
155  // and set errno to ETIMEDOUT. So, we will return FALSE on any error
156  // including a ETIMEDOUT.
157  if (pthread_cond_timedwait(&pSem->cond, &pSem->mtx, &tm)) {
158  pthread_mutex_unlock(&pSem->mtx);
159  return FALSE;
160  }
161  }
162  }
163  break;
164  }
165  pSem->cnt--;
166  pthread_mutex_unlock(&pSem->mtx);
167  return TRUE;
168 }

170 void gfxSemSignal(gfxSem *pSem) {
171  pthread_mutex_lock(&pSem->mtx);
172  if (pSem->cnt < pSem->max) {
173  pSem->cnt++;
174  pthread_cond_signal(&pSem->cond);
175  }
176  pthread_mutex_unlock(&pSem->mtx);
177 }

179 semcount_t gfxSemCounter(gfxSem *pSem) {
180  semcount_t res;
181 
182  // The locking is really only required if obtaining the count is a divisible operation
183  // which it might be on a 8/16 bit processor with a 32 bit semaphore count.
184  pthread_mutex_lock(&pSem->mtx);
185  res = pSem->cnt;
186  pthread_mutex_unlock(&pSem->mtx);
187  return res;
188 }
