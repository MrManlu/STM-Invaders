80 #if GWIN_CONSOLE_ESCSEQ
81  // Convert escape sequences to attributes
82  static bool_t ESCtoAttr(char c, uint8_t *pattr) {
83  uint8_t attr;
84 
85  attr = pattr[0];
86  switch(c) {
87  case '0': case '1': case '2': case '3':
88  case '4': case '5': case '6': case '7':
89  attr &= ~(ESC_REDBIT|ESC_GREENBIT|ESC_BLUEBIT);
90  attr |= (c - '0') | ESC_USECOLOR;
91  break;
92  case 'C':
93  attr &= ~(ESC_REDBIT|ESC_GREENBIT|ESC_BLUEBIT|ESC_USECOLOR);
94  break;
95  case 'u':
96  attr |= ESC_UNDERLINE;
97  break;
98  case 'U':
99  attr &= ~ESC_UNDERLINE;
100  break;
101  case 'b':
102  attr |= ESC_BOLD;
103  break;
104  case 'B':
105  attr &= ~ESC_BOLD;
106  break;
107  default:
108  return FALSE;
109  }
110  if (attr == pattr[0])
111  return FALSE;
112  pattr[0] = attr;
113  return TRUE;
114  }

155  static void HistoryRedraw(GWindowObject *gh) {
156  #define gcw ((GConsoleObject *)gh)
157 
158  // No redrawing if there is no history
159  if (!gcw->buffer)
160  return;
161 
162  // We are printing the buffer - don't store it again
163  gh->flags |= GCONSOLE_FLG_NOSTORE;
164 
165  #if !GWIN_CONSOLE_USE_CLEAR_LINES
166  // Clear the screen
167  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
168  #endif
169 
170  // Reset the cursor
171  gcw->cx = 0;
172  gcw->cy = 0;
173 
174  // Reset the current attributes
175  #if GWIN_CONSOLE_ESCSEQ
176  gcw->currattr = gcw->startattr;
177  #endif
178 
179  // Print the buffer
180  gwinPutCharArray(gh, gcw->buffer, gcw->bufpos);
181 
182  #if GWIN_CONSOLE_USE_CLEAR_LINES
183  // Clear the remaining space
184  {
185  coord_t y;
186 
187  y = gcw->cy;

201  /**
202  * Put a character into our history buffer
203  */
204  static void putCharInBuffer(GConsoleObject *gcw, char c) {
205  // Only store if we need to
206  if (!gcw->buffer || (gcw->g.flags & GCONSOLE_FLG_NOSTORE))
207  return;
208 
209  // Do we have enough space in the buffer
210  if (gcw->bufpos >= gcw->bufsize) {
211  char *p, *ep;
212  size_t dp;
213 
214  /**
215  * This should never really happen except if the user has changed the window
216  * size without turning off and then on the buffer. Even then it is unlikely
217  * because of our conservative allocation strategy.
218  * If it really is needed we scroll one line to make some space. We also mark
219  * it is an overrun so that if asked to really scroll later we know we already have.
220  * Note we only use one bit to indicate an overrun, so an overrun of more
221  * than one line will lead to some interesting scrolling and refreshing
222  * effects.
223  */
224 
225  // Remove one line from the start
226  ep = gcw->buffer+gcw->bufpos;
227  for(p = gcw->buffer; p < ep && *p != '\n'; p++) {
228  #if GWIN_CONSOLE_ESCSEQ
229  if (*p == 27)
230  ESCtoAttr(p[1], &gcw->startattr);
231  #endif
232  }
233 
234  // Was there a newline?
235  if (*p != '\n')
236  p = gcw->buffer; // Oops - no newline, just delete one char
237  else
238  gcw->g.flags |= GCONSOLE_FLG_OVERRUN; // Mark the overrun
239 
240  // Delete the data
241  dp = ++p - gcw->buffer; // Calculate the amount to to be removed
242  gcw->bufpos -= dp; // Calculate the new size
243  if (gcw->bufpos)
244  memcpy(gcw->buffer, p, gcw->bufpos); // Move the rest of the data
245  }
246 
247  // Save the character
248  gcw->buffer[gcw->bufpos++] = c;
249  }

251  /**
252  * Scroll the history buffer by one line
253  */
254  static void scrollBuffer(GConsoleObject *gcw) {
255  char *p, *ep;
256  size_t dp;
257 
258  // Only scroll if we need to
259  if (!gcw->buffer || (gcw->g.flags & GCONSOLE_FLG_NOSTORE))
260  return;
261 
262  // If a buffer overrun has been marked don't scroll as we have already
263  if ((gcw->g.flags & GCONSOLE_FLG_OVERRUN)) {
264  gcw->g.flags &= ~GCONSOLE_FLG_OVERRUN;
265  return;
266  }
267 
268  // Remove one line from the start
269  ep = gcw->buffer+gcw->bufpos;
270  for(p = gcw->buffer; p < ep && *p != '\n'; p++) {
271  #if GWIN_CONSOLE_ESCSEQ
272  if (*p == 27)
273  ESCtoAttr(p[1], &gcw->startattr);
274  #endif
275  }
276 
277  // Was there a newline, if not delete everything.
278  if (*p != '\n') {
279  gcw->bufpos = 0;
280  return;
281  }
282 
283  // Delete the data
284  dp = ++p - gcw->buffer; // Calculate the amount to to be removed
285  gcw->bufpos -= dp; // Calculate the new size
286  if (gcw->bufpos)
287  memcpy(gcw->buffer, p, gcw->bufpos); // Move the rest of the data
288  }

372  #define gcw ((GConsoleObject *)gh)
373 
374  if (gh->vmt != &consoleVMT)
375  return FALSE;

426 void gwinPutChar(GHandle gh, char c) {
427  #define gcw ((GConsoleObject *)gh)
428  uint8_t width, fy;
429 
430  if (gh->vmt != &consoleVMT || !gh->font)
431  return;
432 
433  fy = gdispGetFontMetric(gh->font, fontHeight);
434 
435  #if GWIN_CONSOLE_ESCSEQ
436  /**
437  * Handle escape sequences
438  * ESC color Change subsequent text color
439  * color: "0" = black, "1" = red, "2" = green, "3" = yellow, "4" = blue,
440  * "5" = magenta, "6" = cyan, "7" = white
441  * ESC C Revert subsequent text color to the window default
442  * ESC u Turn on underline
443  * ESC U Turn off underline
444  * ESC b Turn on bold
445  * ESC B Turn off bold
446  * ESC J Clear the window
447  */
448  switch (gcw->escstate) {
449  case 1:
450  gcw->escstate = 0;
451  if (ESCtoAttr(c, &gcw->currattr)) {
452  if (gcw->cx == 0 && gcw->cy == 0)
453  gcw->startattr = gcw->currattr;
454  else {
455  putCharInBuffer(gcw, 27);
456  putCharInBuffer(gcw, c);
457  }
458  } else {
459  switch(c) {
460  case 'J':
461  // Clear the console and reset the cursor
462  clearBuffer(gcw);
463  if (DrawStart(gh)) {
464  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
465  DrawEnd(gh);
466  }
467  gcw->cx = 0;
468  gcw->cy = 0;
469  gcw->startattr = gcw->currattr;
470  break;
471  }
472  }
473  return;
474  }
475  #endif
476 
477  /**
478  * Special Characters:
479  *
480  * Carriage returns and line feeds (\r & \n) are handled in unix terminal cooked mode; that is,
481  * line feeds perform both actions and carriage-returns are ignored.
482  *
483  * if GWIN_CONSOLE_ESCSEQ is turned on then ESC is trapped ready for the escape command.
484  *
485  * All other characters are treated as printable.
486  */
487  switch (c) {
488  case '\n':
489  // clear to the end of the line
490  #if GWIN_CONSOLE_USE_CLEAR_LINES
491  if (gcw->cx == 0 && gcw->cy+fy < gh->height && DrawStart(gh)) {
492  gdispGFillArea(gh->display, gh->x, gh->y + gcw->cy, gh->width, fy, gh->bgcolor);
493  DrawEnd(gh);
494  }
495  #endif
496  // update the cursor
497  gcw->cx = 0;
498  gcw->cy += fy;
499  putCharInBuffer(gcw, '\n');
500  // We use lazy scrolling here and only scroll when the next char arrives
501  return;
502 
503  case '\r':
504  // gcw->cx = 0;
505  return;
506 
507  #if GWIN_CONSOLE_ESCSEQ
508  case 27: // ESC
509  gcw->escstate = 1;
510  return;
511  #endif
512  }
513 
514  // Characters with no width are ignored
515  if (!(width = gdispGetCharWidth(c, gh->font)))
516  return;
517 
518  // Allow space for (very crude) bold
519  #if GWIN_CONSOLE_ESCSEQ
520  if ((gcw->currattr & ESC_BOLD))
521  width++;
522  #endif
523 
524  // Do we need to go to the next line to fit this character?
525  if (gcw->cx + width >= gh->width) {
526  gcw->cx = 0;
527  gcw->cy += fy;
528  putCharInBuffer(gcw, '\n');
529  }
530 
531  // Do we need to scroll to fit this character?
532  if (gcw->cy + fy > gh->height) {
533  #if GWIN_CONSOLE_USE_HISTORY && GWIN_CONSOLE_BUFFER_SCROLLING
534  if (gcw->buffer) {
535  // Scroll the buffer and then redraw using the buffer
536  scrollBuffer(gcw);
537  if (DrawStart(gh)) {
538  HistoryRedraw(gh);
539  DrawEnd(gh);
540  }
541  } else
542  #endif
543  #if GDISP_NEED_SCROLL
544  {
545  // Scroll the console using hardware
546  scrollBuffer(gcw);
547  if (DrawStart(gh)) {
548  gdispGVerticalScroll(gh->display, gh->x, gh->y, gh->width, gh->height, fy, gh->bgcolor);
549  DrawEnd(gh);
550  }
551 
552  // Set the cursor to the start of the last line
553  gcw->cx = 0;
554  gcw->cy = (((coord_t)(gh->height/fy))-1)*fy;
555  }
556  #else
557  {
558  // Clear the console and reset the cursor
559  clearBuffer(gcw);
560  if (DrawStart(gh)) {
561  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
562  DrawEnd(gh);
563  }
564  gcw->cx = 0;
565  gcw->cy = 0;
566  #if GWIN_CONSOLE_ESCSEQ
567  gcw->startattr = gcw->currattr;
568  #endif
569  }
570  #endif
571  }
572 
573  // Save the char
574  putCharInBuffer(gcw, c);
575 
576  // Draw the character
577  if (DrawStart(gh)) {
578 
579  // If we are at the beginning of a new line clear the line
580  #if GWIN_CONSOLE_USE_CLEAR_LINES
581  if (gcw->cx == 0)
582  gdispGFillArea(gh->display, gh->x, gh->y + gcw->cy, gh->width, fy, gh->bgcolor);
583  #endif
584 
585  #if GWIN_CONSOLE_USE_FILLED_CHARS
586  gdispGFillChar(gh->display, gh->x + gcw->cx, gh->y + gcw->cy, c, gh->font, ESCPrintColor(gcw), gh->bgcolor);
587  #else
588  gdispGDrawChar(gh->display, gh->x + gcw->cx, gh->y + gcw->cy, c, gh->font, ESCPrintColor(gcw));
589  #endif
590 
591  #if GWIN_CONSOLE_ESCSEQ
592  // Draw the underline
593  if ((gcw->currattr & ESC_UNDERLINE))
594  gdispGDrawLine(gh->display, gh->x + gcw->cx, gh->y + gcw->cy + fy - gdispGetFontMetric(gh->font, fontDescendersHeight),
595  gh->x + gcw->cx + width + gdispGetFontMetric(gh->font, fontCharPadding), gh->y + gcw->cy + fy - gdispGetFontMetric(gh->font, fontDescendersHeight),
596  ESCPrintColor(gcw));
597  // Bold (very crude)
598  if ((gcw->currattr & ESC_BOLD))
599  gdispGDrawChar(gh->display, gh->x + gcw->cx + 1, gh->y + gcw->cy, c, gh->font, ESCPrintColor(gcw));
600  #endif
601 
602  DrawEnd(gh);
603  }
604 
605  // Update the cursor
606  gcw->cx += width + gdispGetFontMetric(gh->font, fontCharPadding);
607 
608  #undef gcw
609 }
