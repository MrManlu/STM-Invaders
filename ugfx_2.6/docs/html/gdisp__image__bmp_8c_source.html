14 /**
15  * How big a pixel array to allocate for blitting (in pixels)
16  * Bigger is faster but uses more RAM.
17  * This must be greater than 40 bytes and 32 pixels as we read our headers into this space as well
18  */
19 #define BLIT_BUFFER_SIZE_BMP 32

21 typedef struct gdispImagePrivate_BMP {
22  uint8_t bmpflags;
23  #define BMP_V2 0x01 // Version 2 (old) header format
24  #define BMP_V4 0x02 // Version 4 (alpha support) header format
25  #define BMP_PALETTE 0x04 // Uses a palette
26  #define BMP_COMP_RLE 0x08 // Uses RLE compression
27  #define BMP_COMP_MASK 0x10 // Uses mask & shift decoding
28  #define BMP_RLE_ENC 0x20 // Currently in RLE encoded run
29  #define BMP_RLE_ABS 0x40 // Currently in RLE absolute run
30  #define BMP_TOP_TO_BOTTOM 0x80 // Decodes bottom to top line
31  uint8_t bitsperpixel;
32 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
33  uint16_t palsize;
34  pixel_t *palette;
35 #endif
36 #if GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8_RLE
37  uint16_t rlerun;
38  uint8_t rlecode;
39 #endif
40 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_32
41  int8_t shiftred;
42  int8_t shiftgreen;
43  int8_t shiftblue;
44  int8_t shiftalpha;
45  uint32_t maskred;
46  uint32_t maskgreen;
47  uint32_t maskblue;
48  uint32_t maskalpha;
49 #endif
50  size_t frame0pos;
51  pixel_t *frame0cache;
52  pixel_t buf[BLIT_BUFFER_SIZE_BMP];
53  } gdispImagePrivate_BMP;

71 gdispImageError gdispImageOpen_BMP(gdispImage *img) {
72  gdispImagePrivate_BMP *priv;
73  uint8_t hdr[2];
74  uint16_t aword;
75  uint32_t adword;
76  uint32_t offsetColorTable;
77 
78  /* Read the file identifier */
79  if (gfileRead(img->f, hdr, 2) != 2)
80  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
81 
82  /* Process the BITMAPFILEHEADER structure */
83 
84  /**
85  * We only accept Windows V2+ bitmaps.
86  * - we don't support OS/2 bitmaps, icons, pointers, or Windows V1 bitmaps.
87  */
88  if (hdr[0] != 'B' || hdr[1] != 'M')
89  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
90 
91  /* We know we are a BMP format image */
92  img->flags = 0;
93 
94  /* Allocate our private area */
95  if (!(img->priv = gdispImageAlloc(img, sizeof(gdispImagePrivate_BMP))))
96  return GDISP_IMAGE_ERR_NOMEMORY;
97 
98  /* Initialise the essential bits in the private area */
99  priv = (gdispImagePrivate_BMP *)img->priv;
100  priv->frame0cache = 0;
101  priv->bmpflags = 0;
102 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
103  priv->palette = 0;
104 #endif
105 
106  /* Skip the size field and the 2 reserved fields */
107  if (gfileRead(img->f, priv->buf, 8) != 8)
108  goto baddatacleanup;
109 
110  /* Get the offset to the bitmap data */
111  if (gfileRead(img->f, &priv->frame0pos, 4) != 4)
112  goto baddatacleanup;
113  gdispImageMakeLE32(priv->frame0pos);
114 
115  /* Process the BITMAPCOREHEADER structure */
116 
117  /* Get the offset to the colour data */
118  if (gfileRead(img->f, &offsetColorTable, 4) != 4)
119  goto baddatacleanup;
120  gdispImageMakeLE32(offsetColorTable);
121  offsetColorTable += 14; // Add the size of the BITMAPFILEHEADER
122 
123  // Detect our bitmap version
124  if (offsetColorTable == 12+14) {
125  priv->bmpflags |= BMP_V2;
126 
127  // Read the header
128  if (gfileRead(img->f, priv->buf, 12-4) != 12-4)
129  goto baddatacleanup;
130  // Get the width
131  img->width = gdispImageGetAlignedLE16(priv->buf, 0);
132  // Get the height
133  img->height = gdispImageGetAlignedLE16(priv->buf, 2);
134  if (img->height < 0) {
135  priv->bmpflags |= BMP_TOP_TO_BOTTOM;
136  img->height = -img->height;
137  }
138  // Get the planes
139  aword = gdispImageGetAlignedLE16(priv->buf, 4);
140  if (aword != 1)
141  goto unsupportedcleanup;
142  // Get the bits per pixel
143  aword = gdispImageGetAlignedLE16(priv->buf, 6);
144  switch(aword) {
145 #if GDISP_NEED_IMAGE_BMP_1
146  case 1:
147 #endif
148 #if GDISP_NEED_IMAGE_BMP_4
149  case 4:
150 #endif
151 #if GDISP_NEED_IMAGE_BMP_8
152  case 8:
153 #endif
154 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_8
155  priv->bmpflags |= BMP_PALETTE;
156  priv->palsize = 1<<aword;
157  break;
158 #endif
159 #if GDISP_NEED_IMAGE_BMP_24
160  case 24:
161  break;
162 #endif
163  default:
164  goto unsupportedcleanup;
165  }
166  priv->bitsperpixel = aword;
167 
168  } else if (offsetColorTable >= 40+14) {
169  if (offsetColorTable > 40+14)
170  priv->bmpflags |= BMP_V4;
171 
172  // Read the header
173  if (gfileRead(img->f, priv->buf, 40-4) != 40-4)
174  goto baddatacleanup;
175  // Get the width
176  adword = gdispImageGetAlignedLE32(priv->buf, 0);
177  if (adword > 32768) // This also picks up negative values
178  goto unsupportedcleanup;
179  img->width = adword;
180  // Get the height
181  adword = gdispImageGetAlignedLE32(priv->buf, 4);
182  if ((int32_t)adword < 0) { // Negative test
183  priv->bmpflags |= BMP_TOP_TO_BOTTOM;
184  adword = -adword;
185  }
186  if (adword > 32768)
187  goto unsupportedcleanup;
188  img->height = adword;
189  // Get the planes
190  aword = gdispImageGetAlignedLE16(priv->buf, 8);
191  if (aword != 1)
192  goto unsupportedcleanup;
193  // Get the bits per pixel
194  aword = gdispImageGetAlignedLE16(priv->buf, 10);
195  switch(aword) {
196 #if GDISP_NEED_IMAGE_BMP_1
197  case 1:
198 #endif
199 #if GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE
200  case 4:
201 #endif
202 #if GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
203  case 8:
204 #endif
205 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
206  priv->bmpflags |= BMP_PALETTE;
207  priv->palsize = 1<<aword;
208  break;
209 #endif
210 #if GDISP_NEED_IMAGE_BMP_16
211  case 16:
212 #endif
213 #if GDISP_NEED_IMAGE_BMP_24
214  case 24:
215 #endif
216 #if GDISP_NEED_IMAGE_BMP_32
217  case 32:
218 #endif
219 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_24 || GDISP_NEED_IMAGE_BMP_32
220  break;
221 #endif
222  default:
223  goto unsupportedcleanup;
224  }
225  priv->bitsperpixel = aword;
226  // Get the compression
227  adword = gdispImageGetAlignedLE32(priv->buf, 12);
228  switch(adword) {
229  case 0: // BI_RGB - uncompressed
230  break;
231 #if GDISP_NEED_IMAGE_BMP_8_RLE
232  case 1: // BI_RLE8 compression
233  if (priv->bitsperpixel != 8)
234  goto unsupportedcleanup;
235  priv->bmpflags |= BMP_COMP_RLE;
236  break;
237 #endif
238 #if GDISP_NEED_IMAGE_BMP_4_RLE
239  case 2: // BI_RLE4 compression
240  if (priv->bitsperpixel != 4)
241  goto unsupportedcleanup;
242  priv->bmpflags |= BMP_COMP_RLE;
243  break;
244 #endif
245 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_32
246  case 3: // BI_BITFIELDS decoding
247  if (priv->bitsperpixel < 16 || priv->bitsperpixel == 24)
248  goto unsupportedcleanup;
249  priv->bmpflags |= BMP_COMP_MASK;
250  if (priv->bmpflags & BMP_V4) // V4 stored the masks in the header
251  offsetColorTable = 40+14;
252  break;
253 #endif
254  default:
255  goto unsupportedcleanup;
256  }
257  priv->bitsperpixel = aword;
258 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
259  // Get the actual colors used
260  adword = gdispImageGetAlignedLE32(priv->buf, 28);
261  if (adword && adword < priv->palsize)
262  priv->palsize = adword;
263 #endif
264  } else
265  goto baddatacleanup;
266 
267 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
268  /* Load the palette tables */
269  if (priv->bmpflags & BMP_PALETTE) {
270  gfileSetPos(img->f, offsetColorTable);
271 
272  if (!(priv->palette = (color_t *)gdispImageAlloc(img, priv->palsize*sizeof(color_t))))
273  return GDISP_IMAGE_ERR_NOMEMORY;
274  if (priv->bmpflags & BMP_V2) {
275  for(aword = 0; aword < priv->palsize; aword++) {
276  if (gfileRead(img->f, &priv->buf, 3) != 3) goto baddatacleanup;
277  priv->palette[aword] = RGB2COLOR(((uint8_t *)priv->buf)[2], ((uint8_t *)priv->buf)[1], ((uint8_t *)priv->buf)[0]);
278  }
279  } else {
280  for(aword = 0; aword < priv->palsize; aword++) {
281  if (gfileRead(img->f, &priv->buf, 4) != 4) goto baddatacleanup;
282  priv->palette[aword] = RGB2COLOR(((uint8_t *)priv->buf)[2], ((uint8_t *)priv->buf)[1], ((uint8_t *)priv->buf)[0]);
283  }
284  }
285 
286  }
287 #endif
288 
289 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_32
290  /* Load the bit masks */
291  if (priv->bmpflags & BMP_COMP_MASK) {
292  gfileSetPos(img->f, offsetColorTable);
293  if (gfileRead(img->f, &priv->maskred, 4) != 4) goto baddatacleanup;
294  gdispImageMakeLE32(priv->maskred);
295  if (gfileRead(img->f, &priv->maskgreen, 4) != 4) goto baddatacleanup;
296  gdispImageMakeLE32(priv->maskgreen);
297  if (gfileRead(img->f, &priv->maskblue, 4) != 4) goto baddatacleanup;
298  gdispImageMakeLE32(priv->maskblue);
299  if (priv->bmpflags & BMP_V4) {
300  if (gfileRead(img->f, &priv->maskalpha, 4) != 4) goto baddatacleanup;
301  gdispImageMakeLE32(priv->maskalpha);
302  } else
303  priv->maskalpha = 0;
304  } else if (priv->bitsperpixel == 16) {
305  priv->bmpflags |= BMP_COMP_MASK;
306  priv->maskred = 0x7C00;
307  priv->maskgreen = 0x03E0;
308  priv->maskblue = 0x001F;
309  priv->maskalpha = 0;
310  } else if (priv->bitsperpixel == 32) {
311  priv->bmpflags |= BMP_COMP_MASK;
312  priv->maskred = 0x00FF0000;
313  priv->maskgreen = 0x0000FF00;
314  priv->maskblue = 0x000000FF;
315  priv->maskalpha = 0;
316  }
317 
318  /* We need to adjust the masks and calculate the shift values so the result scales 0 -> 255 */
319  if (priv->bmpflags & BMP_COMP_MASK) {
320  priv->shiftred = 0;
321  priv->shiftgreen = 0;
322  priv->shiftblue = 0;
323  if (priv->maskred) {
324  if (priv->maskred < 256)
325  for(adword = priv->maskred; adword < 128; priv->shiftred--, adword <<= 1);
326  else
327  for(adword = priv->maskred; adword > 255; priv->shiftred++, adword >>= 1);
328  }
329  if (priv->maskgreen) {
330  if (priv->maskgreen < 256)
331  for(adword = priv->maskgreen; adword < 128; priv->shiftgreen--, adword <<= 1);
332  else
333  for(adword = priv->maskgreen; adword > 255; priv->shiftgreen++, adword >>= 1);
334  }
335  if (priv->maskblue) {
336  if (priv->maskblue < 256)
337  for(adword = priv->maskblue; adword < 128; priv->shiftblue--, adword <<= 1);
338  else
339  for(adword = priv->maskblue; adword > 255; priv->shiftblue++, adword >>= 1);
340  }
341  if (priv->maskalpha) {
342  if (priv->maskalpha < 256)
343  for(adword = priv->maskalpha; adword < 128; priv->shiftalpha--, adword <<= 1);
344  else
345  for(adword = priv->maskalpha; adword > 255; priv->shiftalpha++, adword >>= 1);
346  }
347  }
348 #endif
349 
350  img->type = GDISP_IMAGE_TYPE_BMP;
351  return GDISP_IMAGE_ERR_OK;
352 
353 baddatacleanup:
354  gdispImageClose_BMP(img); // Clean up the private data area
355  return GDISP_IMAGE_ERR_BADDATA; // Oops - something wrong
356 
357 unsupportedcleanup:
358  gdispImageClose_BMP(img); // Clean up the private data area
359  return GDISP_IMAGE_ERR_UNSUPPORTED; // Not supported
360 }

362 static coord_t getPixels(gdispImage *img, coord_t x) {
363  gdispImagePrivate_BMP * priv;
364  color_t * pc;
365  coord_t len;
366 
367  priv = (gdispImagePrivate_BMP *)img->priv;
368  pc = priv->buf;
369  len = 0;
370 
371  switch(priv->bitsperpixel) {
372 #if GDISP_NEED_IMAGE_BMP_1
373  case 1:
374  {
375  uint8_t b[4];
376  uint8_t m;
377 
378  priv = (gdispImagePrivate_BMP *)img->priv;
379  pc = priv->buf;
380  len = 0;
381 
382  while(x < img->width && len <= BLIT_BUFFER_SIZE_BMP-32) {
383  if (gfileRead(img->f, &b, 4) != 4)
384  return 0;
385 
386  for(m=0x80; m; m >>= 1, pc++)
387  pc[0] = priv->palette[(m&b[0]) ? 1 : 0];
388  for(m=0x80; m; m >>= 1, pc++)
389  pc[0] = priv->palette[(m&b[1]) ? 1 : 0];
390  for(m=0x80; m; m >>= 1, pc++)
391  pc[0] = priv->palette[(m&b[2]) ? 1 : 0];
392  for(m=0x80; m; m >>= 1, pc++)
393  pc[0] = priv->palette[(m&b[3]) ? 1 : 0];
394  len += 32;
395  x += 32;
396  }
397  }
398  return len;
399 #endif
400 
401 #if GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE
402  case 4:
403  #if GDISP_NEED_IMAGE_BMP_4_RLE
404  #if GDISP_NEED_IMAGE_BMP_4
405  if (priv->bmpflags & BMP_COMP_RLE)
406  #endif
407  {
408  uint8_t b[4];
409 
410  while(x < img->width) {
411  if (priv->bmpflags & BMP_RLE_ENC) {
412  while (priv->rlerun && len <= BLIT_BUFFER_SIZE_BMP-2 && x < img->width) {
413  *pc++ = priv->palette[priv->rlecode >> 4];
414  priv->rlerun--;
415  len++;
416  x++;
417  if (priv->rlerun) {
418  *pc++ = priv->palette[priv->rlecode & 0x0F];
419  priv->rlerun--;
420  len++;
421  x++;
422  }
423  }
424  if (priv->rlerun) // Return if we have more run to do
425  return len;
426  } else if (priv->bmpflags & BMP_RLE_ABS) {
427  while (priv->rlerun && len <= BLIT_BUFFER_SIZE_BMP-2 && x < img->width) {
428  if (gfileRead(img->f, &b, 1) != 1)
429  return 0;
430  *pc++ = priv->palette[b[0] >> 4];
431  priv->rlerun--;
432  len++;
433  x++;
434  if (priv->rlerun) {
435  *pc++ = priv->palette[b[0] & 0x0F];
436  priv->rlerun--;
437  len++;
438  x++;
439  }
440  }
441  if (priv->rlerun) // Return if we have more run to do
442  return len;
443  if ((gfileGetPos(img->f) - priv->frame0pos)&1) { // Make sure we are on a word boundary
444  if (gfileRead(img->f, &b, 1) != 1)
445  return 0;
446  }
447  }
448 
449  // We have finished the current run - read a new run
450  priv->bmpflags &= ~(BMP_RLE_ENC|BMP_RLE_ABS);
451 
452  // There are always at least 2 bytes in an RLE code
453  if (gfileRead(img->f, &b, 2) != 2)
454  return 0;
455 
456  if (b[0]) { // Encoded mode
457  priv->rlerun = b[0];
458  priv->rlecode = b[1];
459  priv->bmpflags |= BMP_RLE_ENC;
460  } else if (b[1] == 0) { // End of line
461  if (x < img->width) {
462  priv->rlerun = img->width - x;
463  priv->rlecode = 0; // Who knows what color this should really be
464  priv->bmpflags |= BMP_RLE_ENC;
465  }
466  } else if (b[1] == 1) { // End of file
467  return len;
468  } else if (b[1] == 2) { // Delta x, y
469  // There are always at least 2 bytes in an RLE code
470  if (gfileRead(img->f, &b, 2) != 2)
471  return 0;
472  priv->rlerun = b[0] + (uint16_t)b[1] * img->width;
473  priv->rlecode = 0; // Who knows what color this should really be
474  priv->bmpflags |= BMP_RLE_ENC;
475  } else { // Absolute mode
476  priv->rlerun = b[1];
477  priv->bmpflags |= BMP_RLE_ABS;
478  }
479  }
480  return len;
481  }
482  #endif
483  #if GDISP_NEED_IMAGE_BMP_4
484  {
485  uint8_t b[4];
486 
487  while(x < img->width && len <= BLIT_BUFFER_SIZE_BMP-8) {
488  if (gfileRead(img->f, &b, 4) != 4)
489  return 0;
490 
491  *pc++ = priv->palette[b[0] >> 4];
492  *pc++ = priv->palette[b[0] & 0x0F];
493  *pc++ = priv->palette[b[1] >> 4];
494  *pc++ = priv->palette[b[1] & 0x0F];
495  *pc++ = priv->palette[b[2] >> 4];
496  *pc++ = priv->palette[b[2] & 0x0F];
497  *pc++ = priv->palette[b[3] >> 4];
498  *pc++ = priv->palette[b[3] & 0x0F];
499  len += 8;
500  x += 8;
501  }
502  return len;
503  }
504  #endif
505 #endif
506 
507 #if GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
508  case 8:
509  #if GDISP_NEED_IMAGE_BMP_8_RLE
510  #if GDISP_NEED_IMAGE_BMP_8
511  if (priv->bmpflags & BMP_COMP_RLE)
512  #endif
513  {
514  uint8_t b[4];
515 
516  while(x < img->width) {
517  if (priv->bmpflags & BMP_RLE_ENC) {
518  while (priv->rlerun && len < BLIT_BUFFER_SIZE_BMP && x < img->width) {
519  *pc++ = priv->palette[priv->rlecode];
520  priv->rlerun--;
521  len++;
522  x++;
523  }
524  if (priv->rlerun) // Return if we have more run to do
525  return len;
526  } else if (priv->bmpflags & BMP_RLE_ABS) {
527  while (priv->rlerun && len < BLIT_BUFFER_SIZE_BMP && x < img->width) {
528  if (gfileRead(img->f, &b, 1) != 1)
529  return 0;
530  *pc++ = priv->palette[b[0]];
531  priv->rlerun--;
532  len++;
533  x++;
534  }
535  if (priv->rlerun) // Return if we have more run to do
536  return len;
537  if ((gfileGetPos(img->f) - priv->frame0pos)&1) { // Make sure we are on a word boundary
538  if (gfileRead(img->f, &b, 1) != 1)
539  return 0;
540  }
541  }
542 
543  // We have finished the current run - read a new run
544  priv->bmpflags &= ~(BMP_RLE_ENC|BMP_RLE_ABS);
545 
546  // There are always at least 2 bytes in an RLE code
547  if (gfileRead(img->f, &b, 2) != 2)
548  return 0;
549 
550  if (b[0]) { // Encoded mode
551  priv->rlerun = b[0];
552  priv->rlecode = b[1];
553  priv->bmpflags |= BMP_RLE_ENC;
554  } else if (b[1] == 0) { // End of line
555  if (x < img->width) {
556  priv->rlerun = img->width - x;
557  priv->rlecode = 0; // Who knows what color this should really be
558  priv->bmpflags |= BMP_RLE_ENC;
559  }
560  } else if (b[1] == 1) { // End of file
561  return len;
562  } else if (b[1] == 2) { // Delta x, y
563  // There are always at least 2 bytes in an RLE code
564  if (gfileRead(img->f, &b, 2) != 2)
565  return GDISP_IMAGE_ERR_BADDATA;
566  priv->rlerun = b[0] + (uint16_t)b[1] * img->width;
567  priv->rlecode = 0; // Who knows what color this should really be
568  priv->bmpflags |= BMP_RLE_ENC;
569  } else { // Absolute mode
570  priv->rlerun = b[1];
571  priv->bmpflags |= BMP_RLE_ABS;
572  }
573  }
574  return len;
575  }
576  #endif
577  #if GDISP_NEED_IMAGE_BMP_8
578  {
579  uint8_t b[4];
580 
581  while(x < img->width && len <= BLIT_BUFFER_SIZE_BMP-4) {
582  if (gfileRead(img->f, &b, 4) != 4)
583  return 0;
584 
585  *pc++ = priv->palette[b[0]];
586  *pc++ = priv->palette[b[1]];
587  *pc++ = priv->palette[b[2]];
588  *pc++ = priv->palette[b[3]];
589  len += 4;
590  x += 4;
591  }
592  return len;
593  }

598  case 16:
599  {
600  uint16_t w[2];
601  color_t r, g, b;
602 
603  while(x < img->width && len <= BLIT_BUFFER_SIZE_BMP-2) {
604  if (gfileRead(img->f, &w, 4) != 4)
605  return 0;
606  gdispImageMakeLE16(w[0]);
607  gdispImageMakeLE16(w[1]);
608  if (priv->shiftred < 0)
609  r = (color_t)((w[0] & priv->maskred) << -priv->shiftred);
610  else
611  r = (color_t)((w[0] & priv->maskred) >> priv->shiftred);
612  if (priv->shiftgreen < 0)
613  g = (color_t)((w[0] & priv->maskgreen) << -priv->shiftgreen);
614  else
615  g = (color_t)((w[0] & priv->maskgreen) >> priv->shiftgreen);
616  if (priv->shiftblue < 0)
617  b = (color_t)((w[0] & priv->maskblue) << -priv->shiftblue);
618  else
619  b = (color_t)((w[0] & priv->maskblue) >> priv->shiftblue);
620  /* We don't support alpha yet */
621  *pc++ = RGB2COLOR(r, g, b);
622  if (priv->shiftred < 0)
623  r = (color_t)((w[1] & priv->maskred) << -priv->shiftred);
624  else
625  r = (color_t)((w[1] & priv->maskred) >> priv->shiftred);
626  if (priv->shiftgreen < 0)
627  g = (color_t)((w[1] & priv->maskgreen) << -priv->shiftgreen);
628  else
629  g = (color_t)((w[1] & priv->maskgreen) >> priv->shiftgreen);
630  if (priv->shiftblue < 0)
631  b = (color_t)((w[1] & priv->maskblue) << -priv->shiftblue);
632  else
633  b = (uint8_t)((w[1] & priv->maskblue) >> priv->shiftblue);
634  /* We don't support alpha yet */
635  *pc++ = RGB2COLOR(r, g, b);
636  x += 2;
637  len += 2;
638  }
639  }
640  return len;
641 #endif
642 
643 #if GDISP_NEED_IMAGE_BMP_24
644  case 24:
645  {
646  uint8_t b[3];
647 
648  while(x < img->width && len < BLIT_BUFFER_SIZE_BMP) {
649  if (gfileRead(img->f, &b, 3) != 3)
650  return 0;
651  *pc++ = RGB2COLOR(b[2], b[1], b[0]);
652  x++;
653  len++;
654  }
655 
656  if (x >= img->width) {
657  // Make sure we have read a multiple of 4 bytes for the line
658  if ((x & 3) && gfileRead(img->f, &b, x & 3) != (x & 3))
659  return 0;
660  }
661  }
662  return len;
663 #endif
664 
665 #if GDISP_NEED_IMAGE_BMP_32
666  case 32:
667  {
668  uint32_t dw;
669  color_t r, g, b;
670 
671  while(x < img->width && len < BLIT_BUFFER_SIZE_BMP) {
672  if (gfileRead(img->f, &dw, 4) != 4)
673  return 0;
674  gdispImageMakeLE32(dw);
675  if (priv->shiftred < 0)
676  r = (color_t)((dw & priv->maskred) << -priv->shiftred);
677  else
678  r = (color_t)((dw & priv->maskred) >> priv->shiftred);

684  b = (color_t)((dw & priv->maskblue) << -priv->shiftblue);
685  else
686  b = (color_t)((dw & priv->maskblue) >> priv->shiftblue);
687  /* We don't support alpha yet */
688  *pc++ = RGB2COLOR(r, g, b);
689  x++;
690  len++;
691  }
692  }
693  return len;
694 #endif
695 
696  default:

701 gdispImageError gdispImageCache_BMP(gdispImage *img) {
702  gdispImagePrivate_BMP * priv;
703  color_t * pcs;
704  color_t * pcd;
705  coord_t pos, x, y;
706  size_t len;
707 
708  /* If we are already cached - just return OK */
709  priv = (gdispImagePrivate_BMP *)img->priv;
710  if (priv->frame0cache)
711  return GDISP_IMAGE_ERR_OK;
712 
713  /* We need to allocate the cache */
714  len = img->width * img->height * sizeof(pixel_t);
715  priv->frame0cache = (pixel_t *)gdispImageAlloc(img, len);
716  if (!priv->frame0cache)
717  return GDISP_IMAGE_ERR_NOMEMORY;
718 
719  /* Read the entire bitmap into cache */
720  gfileSetPos(img->f, priv->frame0pos);
721 #if GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8_RLE
722  priv->rlerun = 0;
723  priv->rlecode = 0;
724 #endif
725 
726  pcs = priv->buf; // This line is just to prevent a compiler warning.
727 
728  if (priv->bmpflags & BMP_TOP_TO_BOTTOM) {
729  for(y = 0, pcd = priv->frame0cache; y < img->height; y++) {
730  x = 0; pos = 0;
731  while(x < img->width) {
732  if (!pos) {
733  if (!(pos = getPixels(img, x)))
734  return GDISP_IMAGE_ERR_BADDATA;
735  pcs = priv->buf;
736  }
737  *pcd++ = *pcs++;
738  x++; pos--;
739  }
740  }
741  } else {
742  for(y = img->height-1, pcd = priv->frame0cache + img->width*(img->height-1); y >= 0; y--, pcd -= 2*img->width) {
743  x = 0; pos = 0;
744  while(x < img->width) {
745  if (!pos) {
746  if (!(pos = getPixels(img, x)))
747  return GDISP_IMAGE_ERR_BADDATA;
748  pcs = priv->buf;
749  }
750  *pcd++ = *pcs++;
751  x++; pos--;
752  }
753  }
754  }
755 
756  return GDISP_IMAGE_ERR_OK;
757 }

759 gdispImageError gdispGImageDraw_BMP(GDisplay *g, gdispImage *img, coord_t x, coord_t y, coord_t cx, coord_t cy, coord_t sx, coord_t sy) {
760  gdispImagePrivate_BMP * priv;
761  coord_t mx, my;
762  coord_t pos, len, st;
763 
764  priv = (gdispImagePrivate_BMP *)img->priv;
765 
766  /* Check some reasonableness */
767  if (sx >= img->width || sy >= img->height) return GDISP_IMAGE_ERR_OK;
768  if (sx + cx > img->width) cx = img->width - sx;
769  if (sy + cy > img->height) cy = img->height - sy;
770 
771  /* Draw from the image cache - if it exists */
772  if (priv->frame0cache) {
773  gdispGBlitArea(g, x, y, cx, cy, sx, sy, img->width, priv->frame0cache);
774  return GDISP_IMAGE_ERR_OK;
775  }
776 
777  /* Start decoding from the beginning */
778  gfileSetPos(img->f, priv->frame0pos);
779 #if GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8_RLE
780  priv->rlerun = 0;
781  priv->rlecode = 0;
782 #endif
783 
784  if (priv->bmpflags & BMP_TOP_TO_BOTTOM) {
785  for(my = 0; my < img->height; my++) {
786  mx = 0;
787  while(mx < img->width) {
788  if (!(pos = getPixels(img, mx)))
789  return GDISP_IMAGE_ERR_BADDATA;
790  if (my >= sy && my < sy+cy && mx < sx+cx && mx+pos >= sx) {
791  st = mx < sx ? sx - mx : 0;
792  len = pos-st;
793  if (mx+st+len > sx+cx) len = sx+cx-mx-st;
794  if (len == 1)
795  gdispGDrawPixel(g, x+mx+st-sx, y+my-sy, priv->buf[st]);
796  else
797  gdispGBlitArea(g, x+mx+st-sx, y+my-sy, len, 1, st, 0, pos, priv->buf);
798  }
799  mx += pos;
800  }
801  }
802  } else {
803  for(my = img->height-1; my >= 0; my--) {
804  mx = 0;
805  while(mx < img->width) {
806  if (!(pos = getPixels(img, mx)))
807  return GDISP_IMAGE_ERR_BADDATA;
808  if (my >= sy && my < sy+cy && mx < sx+cx && mx+pos >= sx) {
809  st = mx < sx ? sx - mx : 0;
810  len = pos-st;
811  if (mx+st+len > sx+cx) len = sx+cx-mx-st;
812  if (len == 1)
813  gdispGDrawPixel(g, x+mx+st-sx, y+my-sy, priv->buf[st]);
814  else
815  gdispGBlitArea(g, x+mx+st-sx, y+my-sy, len, 1, st, 0, pos, priv->buf);
816  }
817  mx += pos;
818  }
819  }
820  }
821 
822  return GDISP_IMAGE_ERR_OK;
823 }
