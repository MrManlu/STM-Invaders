29 static bool_t resizeText(GWidgetObject* gw, size_t pos, int32_t diff) {
30  char *p, *q;
31  size_t sz;
32 
33  p = (char *)gw->text;
34  sz = strlen(p)+1;
35  if (diff < 0)
36  memcpy(p+pos, p+pos-diff, sz-pos+diff);
37  if (!(p = gfxRealloc(p, sz, sz+diff)))
38  return FALSE;
39  gw->text = p;
40  if (diff > 0) {
41  q = p + sz;
42  p += pos;
43  while(--q >= p)
44  q[diff] = q[0];
45  }
46  return TRUE;
47 }

54 static void TextEditMouseDown(GWidgetObject* gw, coord_t x, coord_t y) {
55  uint16_t i = 0;
56 
57  (void)y;
58 
59  // Directly jump to the end of the string
60  if (x > gdispGetStringWidth(gw->text, gw->g.font)) {
61  gw2obj->cursorPos = strlen(gw->text);
62 
63  // Otherwise iterate through each character and get the size in pixels to compare
64  } else {
65  i = 1;
66  while (gdispGetStringWidthCount(gw->text, gw->g.font, i) < x) {
67  i++;
68  }
69 
70  gw2obj->cursorPos = i-1;
71  }
72 
73  _gwinUpdate((GHandle)gw);
74 }

76 #if (GFX_USE_GINPUT && GINPUT_NEED_KEYBOARD) || GWIN_NEED_KEYBOARD
77  static void TextEditKeyboard(GWidgetObject* gw, GEventKeyboard* pke) {
78  // Only react on KEYDOWN events. Ignore KEYUP events.
79  if ((pke->keystate & GKEYSTATE_KEYUP) || !pke->bytecount)
80  return;
81 
82  // Is it a special key?
83  if (pke->keystate & GKEYSTATE_SPECIAL) {
84 
85  // Arrow keys to move the cursor
86  switch ((uint8_t)pke->c[0]) {
87  case GKEY_LEFT:
88  if (!gw2obj->cursorPos)
89  return;
90  gw2obj->cursorPos--;
91  break;
92  case GKEY_RIGHT:
93  if (!gw->text[gw2obj->cursorPos])
94  return;
95  gw2obj->cursorPos++;
96  break;
97  case GKEY_HOME:
98  if (!gw2obj->cursorPos)
99  return;
100  gw2obj->cursorPos = 0;
101  break;
102  case GKEY_END:
103  if (!gw->text[gw2obj->cursorPos])
104  return;
105  gw2obj->cursorPos = strlen(gw->text);
106  break;
107  default:
108  return;
109  }
110 
111  } else {
112 
113  // Normal key press
114  switch((uint8_t)pke->c[0]) {
115  case GKEY_BACKSPACE:
116  // Backspace
117  if (!gw2obj->cursorPos)
118  return;
119  gw2obj->cursorPos--;
120  resizeText(gw, gw2obj->cursorPos, -1);
121  break;
122  case GKEY_TAB:
123  case GKEY_LF:
124  case GKEY_CR:
125  // Move to the next field
126  _gwinMoveFocus();
127  return;
128  case GKEY_DEL:
129  // Delete
130  if (!gw->text[gw2obj->cursorPos])
131  return;
132  resizeText(gw, gw2obj->cursorPos, -1);
133  break;
134  default:
135  // Ignore any other control characters
136  if ((uint8_t)pke->c[0] < GKEY_SPACE)
137  return;
138 
139  // Keep the edit length to less than the maximum
140  if (gw2obj->maxSize && gw2obj->cursorPos+pke->bytecount > gw2obj->maxSize)
141  return;
142 
143  // Make space
144  resizeText(gw, gw2obj->cursorPos, pke->bytecount);
145 
146  // Insert the character
147  memcpy((char *)gw->text+gw2obj->cursorPos, pke->c, pke->bytecount);
148  gw2obj->cursorPos += pke->bytecount;
149  break;
150  }
151  }
152 
153  _gwinUpdate((GHandle)gw);
154  }

197 GHandle gwinGTexteditCreate(GDisplay* g, GTexteditObject* wt, GWidgetInit* pInit, size_t maxSize)
198 {
199  char *p;
200 
201  // Create the underlying widget
202  if (!(wt = (GTexteditObject*)_gwidgetCreate(g, &wt->w, pInit, &texteditVMT)))
203  return 0;
204 
205  wt->maxSize = maxSize;
206 
207  // Reallocate the text (if necessary)
208  if (!(wt->w.g.flags & GWIN_FLG_ALLOCTXT)) {
209  if (!(p = gfxAlloc(wt->maxSize+1)))
210  return 0;
211  strncpy(p, wt->w.text, wt->maxSize);
212  wt->w.text = p;
213  wt->w.g.flags |= GWIN_FLG_ALLOCTXT;
214  }
215 
216  // Set text and cursor position
217  wt->cursorPos = strlen(wt->w.text);
218 
219  gwinSetVisible(&wt->w.g, pInit->g.show);
220 
221  return (GHandle)wt;
222 }

224 void gwinTexteditDefaultDraw(GWidgetObject* gw, void* param)
225 {
226  const char* p;
227  coord_t cpos, tpos;
228  const GColorSet* pcol;
229 
230  (void)param;
231 
232  // Is it a valid handle?
233  if (gw->g.vmt != (gwinVMT*)&texteditVMT)
234  return;
235 
236  // Retrieve colors
237  if ((gw->g.flags & GWIN_FLG_SYSENABLED))
238  pcol = &gw->pstyle->enabled;
239  else
240  pcol = &gw->pstyle->disabled;
241 
242  // Adjust the text position so the cursor fits in the window
243  p = gw->text;
244  if (!gw2obj->cursorPos)
245  tpos = 0;
246  else {
247  for(cpos = gw2obj->cursorPos; ; p++, cpos--) {
248  tpos = gdispGetStringWidthCount(p, gw->g.font, cpos);
249  if (tpos < gw->g.width-(TEXT_PADDING_LEFT+CURSOR_PADDING_LEFT))
250  break;
251  }
252  }
253 
254  // Render background and string
255  #if TEXT_PADDING_LEFT
256  gdispGFillArea(gw->g.display, gw->g.x, gw->g.y, TEXT_PADDING_LEFT, gw->g.height, pcol->fill);
257  #endif
258  gdispGFillStringBox(gw->g.display, gw->g.x + TEXT_PADDING_LEFT, gw->g.y, gw->g.width-TEXT_PADDING_LEFT, gw->g.height, p, gw->g.font, pcol->text, pcol->fill, justifyLeft);
259 
260  // Render cursor (if focused)
261  if (gwinGetFocus() == (GHandle)gw) {
262  // Calculate cursor stuff
263 
264  // Draw cursor
265  tpos += gw->g.x + CURSOR_PADDING_LEFT + TEXT_PADDING_LEFT + gdispGetFontMetric(gw->g.font, fontBaselineX)/2;
266  cpos = (gw->g.height - gdispGetFontMetric(gw->g.font, fontHeight))/2 - CURSOR_EXTRA_HEIGHT;
267  gdispGDrawLine(gw->g.display, tpos, gw->g.y + cpos, tpos, gw->g.y + gw->g.height - cpos, pcol->edge);
268  }
269 
270  // Render border
271  gdispGDrawBox(gw->g.display, gw->g.x, gw->g.y, gw->g.width, gw->g.height, pcol->edge);
272 
273  // Render highlighted border if focused
274  _gwidgetDrawFocusRect(gw, 0, 0, gw->g.width, gw->g.height);
275 
276 }
