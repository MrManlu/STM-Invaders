13 static bool is_wrap_space(uint16_t c)
14 {
15  return c == ' ' || c == '\n' || c == '\t' || c == '\r' || c == '-';

20 /* Represents a single word and the whitespace after it. */
21 struct wordlen_s
22 {
23  int16_t word; /* Length of the word in pixels. */
24  int16_t space; /* Length of the whitespace in pixels. */
25  uint16_t chars; /* Number of characters in word + space, combined. */
26 };

28 /* Take the next word from the string and compute its width.
29  * Returns true if the word ends in a linebreak. */
30 static bool get_wordlen(const struct mf_font_s *font, mf_str *text,
31  struct wordlen_s *result)
32 {
33  mf_char c;
34  mf_str prev;
35 
36  result->word = 0;
37  result->space = 0;
38  result->chars = 0;
39 
40  c = mf_getchar(text);
41  while (c && !is_wrap_space(c))
42  {
43  result->chars++;
44  result->word += mf_character_width(font, c);
45  c = mf_getchar(text);
46  }
47 
48  prev = *text;
49  while (c && is_wrap_space(c))
50  {
51  result->chars++;
52 
53  if (c == ' ')
54  result->space += mf_character_width(font, c);
55  else if (c == '\t')
56  result->space += mf_character_width(font, 'm') * MF_TABSIZE;
57  else if (c == '\n')
58  break;
59 
60  prev = *text;
61  c = mf_getchar(text);
62  }
63 
64  /* The last loop reads the first character of next word, put it back. */
65  if (c)
66  *text = prev;
67 
68  return (c == '\0' || c == '\n');
69 }

71 /* Represents the rendered length for a single line. */
72 struct linelen_s
73 {
74  mf_str start; /* Start of the text for line. */
75  uint16_t chars; /* Total number of characters on the line. */
76  int16_t width; /* Total length of all words + whitespace on the line in pixels. */
77  bool linebreak; /* True if line ends in a linebreak */
78  struct wordlen_s last_word; /* Last word on the line. */
79  struct wordlen_s last_word_2; /* Second to last word on the line. */
80 };

82 /* Append word onto the line if it fits. If it would overflow, don't add and
83  * return false. */
84 static bool append_word(const struct mf_font_s *font, int16_t width,
85  struct linelen_s *current, mf_str *text)
86 {
87  mf_str tmp = *text;
88  struct wordlen_s wordlen;
89  bool linebreak;
90 
91  linebreak = get_wordlen(font, &tmp, &wordlen);
92 
93  if (current->width + wordlen.word <= width)
94  {
95  *text = tmp;
96  current->last_word_2 = current->last_word;
97  current->last_word = wordlen;
98  current->linebreak = linebreak;
99  current->chars += wordlen.chars;
100  current->width += wordlen.word + wordlen.space;
101  return true;
102  }
103  else
104  {
105  return false;
106  }
107 }

109 /* Append a character to the line if it fits. */
110 static bool append_char(const struct mf_font_s *font, int16_t width,
111  struct linelen_s *current, mf_str *text)
112 {
113  mf_str tmp = *text;
114  mf_char c;
115  uint16_t w;
116 
117  c = mf_getchar(&tmp);
118  w = mf_character_width(font, c);
119 
120  if (current->width + w <= width)
121  {
122  *text = tmp;
123  current->chars++;
124  current->width += w;
125  return true;
126  }
127  else
128  {
129  return false;
130  }
131 }

135 /* Try to balance the lines by potentially moving one word from the previous
136  * line to the the current one. */
137 static void tune_lines(struct linelen_s *current, struct linelen_s *previous,
138  int16_t max_width)
139 {
140  int16_t curw1, prevw1;
141  int16_t curw2, prevw2;
142  int32_t delta1, delta2;
143 
144  /* If the lines are rendered as is */
145  curw1 = current->width - current->last_word.space;
146  prevw1 = previous->width - previous->last_word.space;
147  delta1 = sq16(max_width - prevw1) + sq16(max_width - curw1);
148 
149  /* If the last word is moved */
150  curw2 = current->width + previous->last_word.word;
151  prevw2 = previous->width - previous->last_word.word
152  - previous->last_word.space
153  - previous->last_word_2.space;
154  delta2 = sq16(max_width - prevw2) + sq16(max_width - curw2);
155 
156  if (delta1 > delta2 && curw2 <= max_width)
157  {
158  /* Do the change. */
159  uint16_t chars;
160 
161  chars = previous->last_word.chars;
162  previous->chars -= chars;
163  current->chars += chars;
164  previous->width -= previous->last_word.word + previous->last_word.space;
165  current->width += previous->last_word.word + previous->last_word.space;
166  previous->last_word = previous->last_word_2;
167 
168  while (chars--) mf_rewind(&current->start);
169  }
170 }

172 void mf_wordwrap(const struct mf_font_s *font, int16_t width,
173  mf_str text, mf_line_callback_t callback, void *state)
174 {
175  struct linelen_s current = { 0 };
176  struct linelen_s previous = { 0 };
177  bool full;
178 
179  current.start = text;
180 
181  while (*text)
182  {
183  full = !append_word(font, width, &current, &text);
184 
185  if (full || current.linebreak)
186  {
187  if (!current.chars)
188  {
189  /* We have a very long word. We must just cut it off at some
190  * point. */
191  while (append_char(font, width, &current, &text));
192  }
193 
194  if (previous.chars)
195  {
196  /* Tune the length and dispatch the previous line. */
197  if (!previous.linebreak && !current.linebreak)
198  tune_lines(&current, &previous, width);
199 
200  if (!callback(previous.start, previous.chars, state))
201  return;
202  }
203 
204  previous = current;
205  current.start = text;
206  current.chars = 0;
207  current.width = 0;
208  current.linebreak = false;
209  current.last_word.word = 0;
210  current.last_word.space = 0;
211  current.last_word.chars = 0;
212  }
213  }
214 
215  /* Dispatch the last lines. */
216  if (previous.chars)
217  {
218  if (!callback(previous.start, previous.chars, state))
219  return;
220  }
221 
222  if (current.chars)
223  callback(current.start, current.chars, state);
224 }

228 void mf_wordwrap(const struct mf_font_s *font, int16_t width,
229  mf_str text, mf_line_callback_t callback, void *state)
230 {
231  mf_str linestart;
232 
233  /* Current line width and character count */
234  int16_t lw_cur = 0, cc_cur = 0;
235 
236  /* Previous wrap point */
237  int16_t cc_prev;
238  mf_str ls_prev;
239 
240  linestart = text;
241 
242  while (*text)
243  {
244  cc_prev = 0;
245  ls_prev = text;
246 
247  while (*text)
248  {
249  mf_char c;
250  int16_t new_width;
251  mf_str tmp;
252 
253  tmp = text;
254  c = mf_getchar(&text);
255  new_width = lw_cur + mf_character_width(font, c);
256 
257  if (c == '\n')
258  {
259  cc_prev = cc_cur + 1;
260  ls_prev = text;
261  break;
262  }
263 
264  if (new_width > width)
265  {
266  text = tmp;
267  break;
268  }
269 
270  cc_cur++;
271  lw_cur = new_width;
272 
273  if (is_wrap_space(c))
274  {
275  cc_prev = cc_cur;
276  ls_prev = text;
277  }
278  }
279 
280  /* Handle unbreakable words */
281  if (cc_prev == 0)
282  {
283  cc_prev = cc_cur;
284  ls_prev = text;
285  }
286 
287  if (!callback(linestart, cc_prev, state))
288  return;
289 
290  linestart = ls_prev;
291  text = linestart;
292  lw_cur = 0;
293  cc_cur = 0;
294  }
295 }
