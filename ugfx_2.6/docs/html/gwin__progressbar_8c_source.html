19 // Reset the display position back to the value predicted by the saved progressbar position
20 static void PBResetDisplayPos(GProgressbarObject *gsw) {
21  if (gsw->w.g.width < gsw->w.g.height)
22  gsw->dpos = gsw->w.g.height-1-((gsw->w.g.height-1)*(gsw->pos-gsw->min))/(gsw->max-gsw->min);
23  else
24  gsw->dpos = ((gsw->w.g.width-1)*(gsw->pos-gsw->min))/(gsw->max-gsw->min);
25 }

97 void gwinProgressbarSetRange(GHandle gh, int min, int max) {
98  #define gsw ((GProgressbarObject *)gh)
99 
100  if (gh->vmt != (gwinVMT *)&progressbarVMT)
101  return;
102 
103  if (min == max) // prevent divide by 0 errors.
104  max++;
105  gsw->min = min;
106  gsw->max = max;
107  gsw->pos = min;
108 
109  PBResetDisplayPos(gsw);
110 
111  #undef gsw
112 }

114 void gwinProgressbarSetPosition(GHandle gh, int pos) {
115  #define gsw ((GProgressbarObject *)gh)
116 
117  if (gh->vmt != (gwinVMT *)&progressbarVMT)
118  return;
119 
120  if (gsw->min <= gsw->max) {

131  _gwinUpdate(gh);
132 
133  #undef gsw

151  #define gsw ((GProgressbarObject *)gh)
152 
153  if (gh->vmt != (gwinVMT *)&progressbarVMT)
154  return;

230 void gwinProgressbarDraw_Std(GWidgetObject *gw, void *param) {
231  #define gsw ((GProgressbarObject *)gw)
232 
233  const GColorSet * pcol;
234  (void) param;
235 
236  if (gw->g.vmt != (gwinVMT *)&progressbarVMT)
237  return;
238 
239  // get the colors right
240  if ((gw->g.flags & GWIN_FLG_SYSENABLED))
241  pcol = &gw->pstyle->enabled;
242  else
243  pcol = &gw->pstyle->disabled;
244 
245  // Vertical progressbar
246  if (gw->g.width < gw->g.height) {
247  if (gsw->dpos != gw->g.height-1)

249  if (gsw->dpos != 0)

251  gdispGDrawBox(gw->g.display, gw->g.x, gw->g.y, gw->g.width, gw->g.height, pcol->edge); // Edge
252  gdispGDrawLine(gw->g.display, gw->g.x, gw->g.y+gsw->dpos, gw->g.x+gw->g.width-1, gw->g.y+gsw->dpos, pcol->edge); // Thumb
253 
254  // Horizontal progressbar
255  } else {
256  if (gsw->dpos != gw->g.width-1)
257  gdispGFillArea(gw->g.display, gw->g.x+gsw->dpos, gw->g.y, gw->g.width-gsw->dpos, gw->g.height, pcol->fill); // Inactive area
258  if (gsw->dpos != 0)
259  gdispGFillArea(gw->g.display, gw->g.x, gw->g.y, gsw->dpos, gw->g.height, pcol->progress); // Active Area
