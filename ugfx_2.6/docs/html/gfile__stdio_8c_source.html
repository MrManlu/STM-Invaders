26 int gstdioSeek(GFILE *f, size_t offset, int origin) {
27  switch(origin) {
28  case SEEK_SET:
29  break;
30  case SEEK_CUR:
31  offset += f->pos;
32  break;
33  case SEEK_END:
34  offset += gfileGetSize(f);
35  break;
36  default:
37  return -1;
38  }
39  return gfileSetPos(f, offset) ? 0 : -1;
40 }
