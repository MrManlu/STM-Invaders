254 static void hline_clip(GDisplay *g) {
255  // Swap the points if necessary so it always goes from x to x1
256  if (g->p.x1 < g->p.x) {
257  g->p.cx = g->p.x; g->p.x = g->p.x1; g->p.x1 = g->p.cx;
258  }
259 
260  // Clipping
261  #if NEED_CLIPPING
262  #if GDISP_HARDWARE_CLIP == HARDWARE_AUTODETECT
263  if (!gvmt(g)->setclip)
264  #endif
265  {
266  if (g->p.y < g->clipy0 || g->p.y >= g->clipy1) return;
267  if (g->p.x < g->clipx0) g->p.x = g->clipx0;
268  if (g->p.x1 >= g->clipx1) g->p.x1 = g->clipx1 - 1;
269  if (g->p.x1 < g->p.x) return;
270  }
271  #endif
272

274  // have hardware fills or if we support both hardware pixel drawing and hardware streaming
275  #if GDISP_HARDWARE_FILLS || (GDISP_HARDWARE_DRAWPIXEL && GDISP_HARDWARE_STREAM_WRITE)
276  // Is this a point
277  if (g->p.x == g->p.x1) {
278  drawpixel(g);
279  return;
280  }
281  #endif

284  #if GDISP_HARDWARE_FILLS
285  #if GDISP_HARDWARE_FILLS == HARDWARE_AUTODETECT
286  if (gvmt(g)->fill)
287  #endif
288  {
289  g->p.cx = g->p.x1 - g->p.x + 1;
290  g->p.cy = 1;
291  gdisp_lld_fill_area(g);
292  return;
293  }
294  #endif

297  #if GDISP_HARDWARE_FILLS != TRUE && GDISP_HARDWARE_STREAM_POS && GDISP_HARDWARE_STREAM_WRITE
298  #if GDISP_HARDWARE_STREAM_POS == HARDWARE_AUTODETECT
299  if (gvmt(g)->writepos)
300  #endif

302  if (!(g->flags & GDISP_FLG_SCRSTREAM))
303  setglobalwindow(g);
304  g->p.cx = g->p.x1 - g->p.x + 1;
305  gdisp_lld_write_pos(g);

308  }
309  #endif

319  gdisp_lld_write_start(g);

425 // Parameters: x,y and x1,y1
426 // Alters: x,y x1,y1 cx,cy
427 static void line_clip(GDisplay *g) {
428  int16_t dy, dx;
429  int16_t addx, addy;
430  int16_t P, diff, i;
431 
432  // Is this a horizontal line (or a point)
433  if (g->p.y == g->p.y1) {
434  hline_clip(g);
435  return;
436  }
437 
438  // Is this a vertical line (or a point)
439  if (g->p.x == g->p.x1) {
440  vline_clip(g);
441  return;
442  }
443 
444  // Not horizontal or vertical
445 
446  // Use Bresenham's line drawing algorithm.
447  // This should be replaced with fixed point slope based line drawing
448  // which is more efficient on modern processors as it branches less.
449  // When clipping is needed, all the clipping could also be done up front
450  // instead of on each pixel.
451 
452  if (g->p.x1 >= g->p.x) {
453  dx = g->p.x1 - g->p.x;
454  addx = 1;
455  } else {
456  dx = g->p.x - g->p.x1;
457  addx = -1;
458  }
459  if (g->p.y1 >= g->p.y) {
460  dy = g->p.y1 - g->p.y;
461  addy = 1;
462  } else {
463  dy = g->p.y - g->p.y1;
464  addy = -1;
465  }
466 
467  if (dx >= dy) {
468  dy <<= 1;
469  P = dy - dx;
470  diff = P - dx;
471 
472  for(i=0; i<=dx; ++i) {
473  drawpixel_clip(g);
474  if (P < 0) {
475  P += dy;
476  g->p.x += addx;
477  } else {
478  P += diff;
479  g->p.x += addx;
480  g->p.y += addy;
481  }
482  }
483  } else {
484  dx <<= 1;
485  P = dx - dy;
486  diff = P - dy;
487 
488  for(i=0; i<=dy; ++i) {
489  drawpixel_clip(g);
490  if (P < 0) {
491  P += dx;
492  g->p.y += addy;
493  } else {
494  P += diff;
495  g->p.x += addx;
496  g->p.y += addy;
497  }
498  }
499  }

2907  /* Find a vector (nx, ny) that is perpendicular to (dx, dy) and has length
2908  * equal to 'norm'. */
2909  static void get_normal_vector(coord_t dx, coord_t dy, coord_t norm, coord_t *nx, coord_t *ny)
2910  {
2911  int32_t dx2, dy2, len_sq, norm_sq, norm_sq2;
2912  int div, step, best, delta, abs_delta;
2913 
2914  dx2 = dx; dy2 = dy;
2915  norm_sq = (int32_t)norm * norm;
2916  norm_sq2 = norm_sq * 512;
2917 
2918  /* Scale dx2 and dy2 so that
2919  * len_sq / 2 <= norm_sq * 512 <= len_sq * 2.
2920  * The scaling by 512 is to yield higher accuracy in division later. */
2921  len_sq = dx2 * dx2 + dy2 * dy2;
2922 
2923  if (len_sq < norm_sq2)
2924  {
2925  while (len_sq && len_sq < norm_sq2)
2926  {
2927  len_sq <<= 2; dx2 <<= 1; dy2 <<= 1;
2928  }
2929  }
2930  else if (len_sq > norm_sq2)
2931  {
2932  while (len_sq && len_sq > norm_sq2)
2933  {
2934  len_sq >>= 2; dx2 >>= 1; dy2 >>= 1;
2935  }
2936  }
2937 
2938  /* Now find the divider div so that
2939  * len_sq / div^2 == norm_sq i.e. div = sqrt(len_sq / norm_sq)
2940  *
2941  * This is done using bisection search to avoid the need for floating
2942  * point sqrt.
2943  *
2944  * Based on previous scaling, we know that
2945  * len_sq / 2 <= norm_sq * 512 <=> div <= sqrt(1024) = 32
2946  * len_sq * 2 >= norm_sq * 512 <=> div >= sqrt(256) = 16
2947  */
2948  div = 24; step = 8;
2949  best = 256;
2950 
2951  for (;;)
2952  {
2953  dx = dx2 / div;
2954  dy = dy2 / div;
2955  len_sq = dx*dx + dy*dy;
2956 
2957  delta = len_sq - norm_sq;
2958 
2959  abs_delta = (delta >= 0) ? delta : -delta;
2960 
2961  if (abs_delta < best)
2962  {
2963  *nx = dy;
2964  *ny = -dx;
2965  best = abs_delta;
2966  }
2967 
2968  if (delta > 0)
2969  div += step;
2970  else if (delta < 0)
2971  div -= step;
2972  else if (delta == 0)
2973  break;
2974 
2975  if (step == 0)
2976  break;
2977  else
2978  step >>= 1; /* Do one round with step = 0 to calculate final result. */
2979  }
