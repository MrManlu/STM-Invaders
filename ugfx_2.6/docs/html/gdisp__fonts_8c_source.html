17 /**
18  * Match a pattern against the font name.
19  */
20 static bool_t matchfont(const char *pattern, const char *name) {
21  while(1) {
22  switch (pattern[0]) {
23  case '*':
24  if (name[0] == 0)
25  return pattern[1] == 0;
26  if (pattern[1] == name[0])
27  pattern++;
28  else
29  name++;
30  break;
31  case 0:
32  return name[0] == 0;
33  default:
34  if (name[0] != pattern[0])
35  return FALSE;
36  pattern++;
37  name++;
38  break;
39  }
40  }
41 }

43 font_t gdispOpenFont(const char *name) {
44  const struct mf_font_list_s *fp;
45 
46 
47  // Try the long names first
48  for(fp = mf_get_font_list(); fp; fp = fp->next) {
49  if (matchfont(name, fp->font->full_name))
50  return fp->font;
51  }
52 
53  // Try the short names if no long names match
54  for(fp = mf_get_font_list(); fp; fp = fp->next) {
55  if (matchfont(name, fp->font->short_name))
56  return fp->font;
57  }
58 
59  /* Return default font.. better than nothing. */
60  return mf_get_font_list()->font;
61 }
