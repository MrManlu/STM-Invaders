32 static void ImageRedraw(GHandle gh) {
33  coord_t x, y, w, h, dx, dy;
34  color_t bg;
35  #if GWIN_NEED_IMAGE_ANIMATION
36  delaytime_t delay;
37  #endif
38 
39  // The default display area
40  dx = 0;
41  dy = 0;
42  x = gh->x;
43  y = gh->y;
44  w = gh->width;
45  h = gh->height;
46  bg = gwinGetDefaultBgColor();
47 
48  // If the image isn't open just clear the area
49  if (!gdispImageIsOpen(&gw->image)) {
50  gdispGFillArea(gh->display, x, y, w, h, bg);
51  return;
52  }
53 
54  // Center horizontally if the area is larger than the image
55  if (gw->image.width < w) {
56  w = gw->image.width;
57  dx = (gh->width-w)/2;
58  x += dx;
59  if (dx)
60  gdispGFillArea(gh->display, gh->x, y, dx, h, bg);
61  gdispGFillArea(gh->display, x+w, y, gh->width-dx-w, h, bg);
62  dx = 0;
63  }
64 
65  // Center image horizontally if the area is smaller than the image
66  else if (gw->image.width > w) {
67  dx = (gw->image.width - w)/2;
68  }
69 
70  // Center vertically if the area is larger than the image
71  if (gw->image.height < h) {
72  h = gw->image.height;
73  dy = (gh->height-h)/2;
74  y += dy;
75  if (dy)
76  gdispGFillArea(gh->display, x, gh->y, w, dy, bg);
77  gdispGFillArea(gh->display, x, y+h, w, gh->height-dy-h, bg);
78  dy = 0;
79  }
80 
81  // Center image vertically if the area is smaller than the image
82  else if (gw->image.height > h) {
83  dy = (gw->image.height - h)/2;
84  }
85 
86  // Reset the background color in case it has changed
87  gdispImageSetBgColor(&gw->image, bg);
88 
89  // Display the image
90  gdispGImageDraw(gh->display, &gw->image, x, y, w, h, dx, dy);
91 
92  #if GWIN_NEED_IMAGE_ANIMATION
93  // read the delay for the next frame
94  delay = gdispImageNext(&gw->image);
95 
96  // Wait for that delay if required
97  switch(delay) {
98  case TIME_INFINITE:
99  // Everything is done
100  break;
101  case TIME_IMMEDIATE:
102  // We can't allow a continuous loop here as it would lock the system up so we delay for the minimum period
103  delay = 1;
104  // Fall through
105  default:
106  // Start the timer to draw the next frame of the animation
107  gtimerStart(&gw->timer, ImageTimer, (void*)gh, FALSE, delay);
108  break;
109  }
110  #endif
111 }

138 bool_t gwinImageOpenGFile(GHandle gh, GFILE *f) {
139  // is it a valid handle?
140  if (gh->vmt != (gwinVMT *)&imageVMT)
141  return FALSE;
142 
143  if (gdispImageIsOpen(&gw->image))
144  gdispImageClose(&gw->image);
145 
146  if ((gdispImageOpenGFile(&gw->image, f) & GDISP_IMAGE_ERR_UNRECOVERABLE))
147  return FALSE;
148 
149  _gwinUpdate(gh);
150
