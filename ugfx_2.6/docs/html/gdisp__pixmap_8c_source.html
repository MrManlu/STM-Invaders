52 GDisplay *gdispPixmapCreate(coord_t width, coord_t height) {
53  GDisplay *g;
54  pixmap *p;
55  unsigned i;
56 
57  // Calculate the size of the display surface in bytes
58  i = width*height*sizeof(color_t);
59  if (i < 2*sizeof(coord_t))
60  i = 2*sizeof(coord_t);
61 
62  // Allocate the pixmap
63  if (!(p = gfxAlloc(i+sizeof(pixmap)-sizeof(p->pixels))))
64  return 0;
65 
66  // Fill in the image header (if required)
67  #if GDISP_NEED_PIXMAP_IMAGE
68  p->imghdr[0] = 'N';
69  p->imghdr[1] = 'I';
70  p->imghdr[2] = (uint8_t)(width >> 8);
71  p->imghdr[3] = (uint8_t)width;
72  p->imghdr[4] = (uint8_t)(height >> 8);
73  p->imghdr[5] = (uint8_t)height;
74  p->imghdr[6] = (uint8_t)(GDISP_PIXELFORMAT >> 8);
75  p->imghdr[7] = (uint8_t)(GDISP_PIXELFORMAT);
76  #endif
77 
78  // Save the width and height so the driver can retrieve it.
79  ((coord_t *)p->pixels)[0] = width;
80  ((coord_t *)p->pixels)[1] = height;
81 
82  // Register the driver
83  g = (GDisplay *)gdriverRegister(&GDISPVMT_pixmap->d, p);
84  if (!g)
85  gfxFree(p);
86  return g;
87 }

113 LLDSPEC bool_t gdisp_lld_init(GDisplay *g) {
114  // The user api function should have already allocated and initialised the pixmap
115  // structure and put it into the priv member during driver initialisation.
116  if (!g->priv)
117  return FALSE;
118 
119  // Initialize the GDISP structure
120  // Width and height were saved into the start of the framebuffer.
121  g->g.Width = ((coord_t *)((pixmap *)g->priv)->pixels)[0];
122  g->g.Height = ((coord_t *)((pixmap *)g->priv)->pixels)[1];
123  g->g.Backlight = 100;
124  g->g.Contrast = 50;
125  g->g.Orientation = GDISP_ROTATE_0;
126  g->g.Powermode = powerOn;
127  g->board = 0;
128 
129  return TRUE;
130 }

136 LLDSPEC void gdisp_lld_draw_pixel(GDisplay *g) {
137  unsigned pos;
138 
139  #if GDISP_NEED_CONTROL
140  switch(g->g.Orientation) {
141  case GDISP_ROTATE_0:
142  default:
143  pos = g->p.y * g->g.Width + g->p.x;
144  break;
145  case GDISP_ROTATE_90:
146  pos = (g->g.Width-g->p.x-1) * g->g.Height + g->p.y;
147  break;
148  case GDISP_ROTATE_180:
149  pos = (g->g.Height-g->p.y-1) * g->g.Width + g->g.Width-g->p.x-1;
150  break;
151  case GDISP_ROTATE_270:
152  pos = g->p.x * g->g.Height + g->g.Height-g->p.y-1;
153  break;
154  }
155  #else
156  pos = g->p.y * g->g.Width + g->p.x;
157  #endif
158 
159  ((pixmap *)(g)->priv)->pixels[pos] = g->p.color;
160 }
